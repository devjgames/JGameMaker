001package org.game;
002
003public class Frustum {
004
005    public static void normalize(Vec4 plane) {
006        float length = (float)Math.sqrt(plane.x * plane.x + plane.y * plane.y + plane.z * plane.z);
007
008        plane.div(length);
009    }
010
011    private final Vec4[] planes = new Vec4[] {
012        new Vec4(),
013        new Vec4(),
014        new Vec4(),
015        new Vec4(),
016        new Vec4(),
017        new Vec4()
018    };
019    private final Mat4 matrix = new Mat4();
020    private final AABB bounds = new AABB();
021
022    public void calcPlanes(Mat4 projection, Mat4 view) {
023        matrix.set(projection).mul(view);
024        matrix.transpose();
025
026        planes[0].set(
027            matrix.m03 + matrix.m00,
028            matrix.m13 + matrix.m10,
029            matrix.m23 + matrix.m20,
030            matrix.m33 + matrix.m30
031        );
032
033        planes[1].set(
034            matrix.m03 - matrix.m00,
035            matrix.m13 - matrix.m10,
036            matrix.m23 - matrix.m20,
037            matrix.m33 - matrix.m30
038        );
039
040        planes[2].set(
041            matrix.m03 + matrix.m01,
042            matrix.m13 + matrix.m11,
043            matrix.m23 + matrix.m21,
044            matrix.m33 + matrix.m31
045        );
046
047        planes[3].set(
048            matrix.m03 - matrix.m01,
049            matrix.m13 - matrix.m11,
050            matrix.m23 - matrix.m21,
051            matrix.m33 - matrix.m31
052        );
053
054        planes[4].set(
055            matrix.m03 + matrix.m02,
056            matrix.m13 + matrix.m12,
057            matrix.m23 + matrix.m22,
058            matrix.m33 + matrix.m32
059        );
060
061        planes[5].set(
062            matrix.m03 - matrix.m02,
063            matrix.m13 - matrix.m12,
064            matrix.m23 - matrix.m22,
065            matrix.m33 - matrix.m32
066        );
067
068        for(Vec4 plane : planes) {
069            normalize(plane);
070        }
071    }
072
073    public boolean contains(AABB b) {
074        if(b.isEmpty()) {
075            return false;
076        }
077        for(Vec4 plane : planes) {
078            float minD = plane.w;
079            float maxD = plane.w;
080
081            minD += (plane.x < 0) ? b.max.x * plane.x : b.min.x * plane.x;
082            minD += (plane.y < 0) ? b.max.y * plane.y : b.min.y * plane.y;
083            minD += (plane.z < 0) ? b.max.z * plane.z : b.min.z * plane.z;
084
085            maxD += (plane.x > 0) ? b.max.x * plane.x : b.min.x * plane.x;
086            maxD += (plane.y > 0) ? b.max.y * plane.y : b.min.y * plane.y;
087            maxD += (plane.z > 0) ? b.max.z * plane.z : b.min.z * plane.z;
088
089            if(minD < 0 && maxD < 0) {
090                return false;
091            }
092        }
093        return true;
094    }
095
096    public boolean contains(Vec3 center, float radius) {
097        return contains(
098            bounds.set(
099                center.x - radius, center.y - radius, center.z - radius,
100                center.x + radius, center.y + radius, center.z + radius
101            )
102        );
103    }
104}
