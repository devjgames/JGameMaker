001package org.game;
002
003public final class AABB {
004    
005    public final Vec3 min = new Vec3();
006    public final Vec3 max = new Vec3();
007
008    public AABB() {
009        clear();
010    }
011
012    public AABB(float x1, float y1, float z1, float x2, float y2, float z2) {
013        set(x1, y1, z1, x2, y2, z2);
014    }
015
016    public AABB(Vec3 min, Vec3 max) {
017        set(min, max);
018    }
019
020    public AABB(AABB b) {
021        set(b);
022    }
023
024    public boolean isEmpty() {
025        return min.x > max.x || min.y > max.y || min.z > max.z;
026    }
027
028    public boolean contains(float x, float y, float z) {
029        if(!isEmpty()) {
030            return x >= min.x && x <= max.x && y >= min.y && y <= max.y && z >= min.z && z <= max.z;
031        }
032        return false;
033    }
034
035    public boolean contains(Vec3 point) {
036        return contains(point.x, point.y, point.z);
037    }
038
039    public boolean touches(AABB b) {
040        if(!isEmpty() && !b.isEmpty()) {
041            return !(
042                b.min.x > max.x || b.max.x < min.x ||
043                b.min.y > max.y || b.max.y < min.y ||
044                b.min.z > max.z || b.max.z < min.z
045            );
046        }
047        return false;
048    }
049
050    public Vec3 center(Vec3 center) {
051        return max.add(min, center).div(2);
052    }
053
054    public Vec3 size(Vec3 size) {
055        return max.sub(min, size);
056    }
057
058    public AABB clear() {
059        min.set(1, 1, 1).scale(Float.MAX_VALUE);
060        min.negate(max);
061
062        return this;
063    }
064
065    public AABB set(float x1, float y1, float z1, float x2, float y2, float z2) {
066        min.set(x1, y1, z1);
067        max.set(x2, y2, z2);
068
069        return this;
070    }
071
072    public AABB set(Vec3 min, Vec3 max) {
073        this.min.set(min);
074        this.max.set(max);
075
076        return this;
077    }
078
079    public AABB set(AABB b) {
080        min.set(b.min);
081        max.set(b.max);
082
083        return this;
084    }
085
086    public AABB buffer(float x, float y, float z) {
087        if(!isEmpty()) {
088            min.sub(x, y, z);
089            max.add(x, y, z);
090        }
091        return this;
092    }
093
094    public AABB buffer(Vec3 amount) {
095        return buffer(amount.x, amount.y, amount.z);
096    }
097    
098    public AABB add(float x, float y, float z) {
099        min.x = Math.min(x, min.x);
100        min.y = Math.min(y, min.y);
101        min.z = Math.min(z, min.z);
102        max.x = Math.max(x, max.x);
103        max.y = Math.max(y, max.y);
104        max.z = Math.max(z, max.z);
105
106        return this;
107    }
108
109    public AABB add(Vec3 point) {
110        return add(point.x, point.y, point.z);
111    }
112
113    public AABB add(AABB b) {
114        if(!b.isEmpty()) {
115            add(b.min);
116            add(b.max);
117        }
118        return this;
119    }
120
121    public AABB transform(Mat4 m) {
122        if(!isEmpty()) {
123            float minx = m.m03;
124            float miny = m.m13;
125            float minz = m.m23;
126            float maxx = minx;
127            float maxy = miny;
128            float maxz = minz;
129
130            minx += (m.m00 < 0) ? m.m00 * max.x : m.m00 * min.x;
131            minx += (m.m01 < 0) ? m.m01 * max.y : m.m01 * min.y;
132            minx += (m.m02 < 0) ? m.m02 * max.z : m.m02 * min.z;
133            maxx += (m.m00 > 0) ? m.m00 * max.x : m.m00 * min.x;
134            maxx += (m.m01 > 0) ? m.m01 * max.y : m.m01 * min.y;
135            maxx += (m.m02 > 0) ? m.m02 * max.z : m.m02 * min.z;
136
137            miny += (m.m10 < 0) ? m.m10 * max.x : m.m10 * min.x;
138            miny += (m.m11 < 0) ? m.m11 * max.y : m.m11 * min.y;
139            miny += (m.m12 < 0) ? m.m12 * max.z : m.m12 * min.z;
140            maxy += (m.m10 > 0) ? m.m10 * max.x : m.m10 * min.x;
141            maxy += (m.m11 > 0) ? m.m11 * max.y : m.m11 * min.y;
142            maxy += (m.m12 > 0) ? m.m12 * max.z : m.m12 * min.z;
143
144            minz += (m.m20 < 0) ? m.m20 * max.x : m.m20 * min.x;
145            minz += (m.m21 < 0) ? m.m21 * max.y : m.m21 * min.y;
146            minz += (m.m22 < 0) ? m.m22 * max.z : m.m22 * min.z;
147            maxz += (m.m20 > 0) ? m.m20 * max.x : m.m20 * min.x;
148            maxz += (m.m21 > 0) ? m.m21 * max.y : m.m21 * min.y;
149            maxz += (m.m22 > 0) ? m.m22 * max.z : m.m22 * min.z;
150            
151            set(minx, miny, minz, maxx, maxy, maxz);
152        }
153        return this;
154    }
155
156    public boolean isects(Vec3 origin, Vec3 direction, float[] time) {
157        float tnear = -Float.MAX_VALUE;
158        float tfar = Float.MAX_VALUE;
159
160        for(int i = 0; i != 3; i++) {
161            if(Math.abs(direction.component(i)) < 0.0000001) {
162                if(origin.component(i) < min.component(i) || origin.component(i) > max.component(i)) {
163                    return false;
164                }
165            } else {
166                float t1 = (min.component(i) - origin.component(i)) / direction.component(i);
167                float t2 = (max.component(i) - origin.component(i)) / direction.component(i);
168                
169                if(t1 > t2) {
170                    float temp = t1;
171                    t1 = t2;
172                    t2 = temp;
173                }
174                if(t1 > tnear) {
175                    tnear = t1;
176                }
177                if(t2 < tfar) {
178                    tfar = t2;
179                }
180                if(tnear > tfar) {
181                    return false;
182                }
183                if(tfar < 0) {
184                    return false;
185                }
186            }
187        }
188        if(tnear > 0) {
189            if(tnear < time[0]) {
190                time[0] = tnear;
191                return true;
192            }
193        } else if(tfar < time[0]) {
194            time[0] = tfar;
195            return true;
196        }
197        return false;
198    }
199
200    @Override
201    public String toString() {
202        return min + " -> " + max;
203    }
204}
