001package org.game;
002
003public final class Vec2 {
004   
005    public float x;
006    public float y;
007
008    public Vec2() {
009        toZero();
010    }
011
012    public Vec2(float x, float y) {
013        set(x, y);
014    }
015
016    public Vec2(Vec2 v) {
017        set(v);
018    }
019
020    public float component(int i) {
021        if(i == 0) {
022            return x;
023        } else {
024            return y;
025        }
026    }
027
028    public void setComponent(int i, float value) {
029        if(i == 0) {
030            x = value;
031        } else {
032            y = value;
033        } 
034    }
035
036    public float lengthSquared() {
037        return x * x + y * y ;
038    }
039
040    public float length() {
041        return (float)Math.sqrt(lengthSquared());
042    }
043
044    public float distance(float x, float y) {
045        float dx = x - this.x;
046        float dy = y - this.y;
047        
048        return (float)Math.sqrt(dx * dx + dy * dy);
049    }
050
051    public float distance(Vec2 v) {
052        return distance(v.x, v.y);
053    }
054
055    public float dot(float x, float y) {
056        return this.x * x + this.y * y;
057    }
058
059    public float dot(Vec2 v) {
060        return dot(v.x, v.y);
061    }
062
063    public Vec2 normalize(float length) {
064        return div(length()).scale(length);
065    }
066
067    public Vec2 normalize() {
068        return normalize(1);
069    }
070
071    public Vec2 normalize(float length, Vec2 dest) {
072        return dest.set(this).normalize(length);
073    }
074
075    public Vec2 normalize(Vec2 dest) {
076        return normalize(1, dest);
077    }
078
079    public Vec2 lerp(Vec2 v, float amount, Vec2 dest) {
080        dest.x = x + amount * (v.x - x);
081        dest.y = y + amount * (v.y - y);
082
083        return dest;
084    }
085
086    public Vec2 lerp(Vec2 v, float amount) {
087        return lerp(v, amount, this);
088    }
089
090    public Vec2 toZero() {
091        x = 0;
092        y = 0;
093
094        return this;
095    }
096
097    public Vec2 set(float x, float y) {
098        this.x = x;
099        this.y = y;
100
101        return this;
102    }
103
104    public Vec2 set(Vec2 v) {
105        x = v.x;
106        y = v.y;
107
108        return this;
109    }
110
111    public Vec2 add(float x, float y, Vec2 dest) {
112        dest.x = this.x + x;
113        dest.y = this.y + y;
114
115        return dest;
116    }
117
118    public Vec2 add(float x, float y) {
119        return add(x, y, this);
120    }
121
122    public Vec2 add(Vec2 v) {
123        return add(v, this);
124    }
125
126    public Vec2 add(Vec2 v, Vec2 dest) {
127        return add(v.x, v.y, dest);
128    }
129
130    public Vec2 sub(float x, float y, Vec2 dest) {
131        dest.x = this.x - x;
132        dest.y = this.y - y;
133
134        return dest;
135    }
136
137    public Vec2 sub(float x, float y) {
138        return sub(x, y, this);
139    }
140
141    public Vec2 sub(Vec2 v) {
142        return sub(v, this);
143    }
144
145    public Vec2 sub(Vec2 v, Vec2 dest) {
146        return sub(v.x, v.y, dest);
147    }
148
149    public Vec2 negate(Vec2 dest) {
150        dest.x = -x;
151        dest.y = -y;
152
153        return dest;
154    }
155
156    public Vec2 negate() {
157        return negate(this);
158    }
159
160    public Vec2 scale(float x, float y, Vec2 dest) {
161        dest.x = this.x * x;
162        dest.y = this.y * y;
163
164        return dest;
165    }
166
167    public Vec2 scale(float x, float y) {
168        return scale(x, y, this);
169    }
170
171    public Vec2 scale(Vec2 v) {
172        return scale(v.x, v.y);
173    }
174
175    public Vec2 scale(float s) {
176        return scale(s, s);
177    }
178
179    public Vec2 scale(float s, Vec2 dest) {
180        return scale(s, s, dest);
181    }
182
183    public Vec2 div(float x, float y, Vec2 dest) {
184        dest.x = this.x / x;
185        dest.y = this.y / y;
186
187        return dest;
188    }
189
190    public Vec2 div(float x, float y) {
191        return div(x, y, this);
192    }
193
194    public Vec2 div(Vec2 v) {
195        return div(v.x, v.y);
196    }
197
198    public Vec2 div(Vec2 v, Vec2 dest) {
199        return div(v.x, v.y, dest);
200    }
201
202    public Vec2 div(float d) {
203        return div(d, d);
204    }
205
206    public Vec2 div(float d, Vec2 dest) {
207        return div(d, d, dest);
208    }
209
210    @Override
211    public String toString() {
212        return x + " " + y;
213    }
214
215    public static Vec2 parse(String s, Vec2 v) {
216        String[] tokens = s.split("\\s+");
217
218        if(tokens.length >= 2) {
219            try {
220                float x = Float.parseFloat(tokens[0]);
221                float y = Float.parseFloat(tokens[1]);
222
223                v.set(x, y);
224            } catch(NumberFormatException ex) {
225            }
226        }
227        return v;
228    }
229}
