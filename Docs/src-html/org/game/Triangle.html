001package org.game;
002
003public final class Triangle {
004    
005    public final Vec3 p1 = new Vec3();
006    public final Vec3 p2 = new Vec3();
007    public final Vec3 p3 = new Vec3();
008    public final Vec3 n = new Vec3();
009    public float d  = 0;
010    public int tag = 0;
011
012    private final Vec3 n2 = new Vec3();
013    private final Vec3 v = new Vec3();
014    private final Vec3 a = new Vec3();
015    private final Vec3 b = new Vec3();
016    private final Vec3 p = new Vec3();
017    private final Vec3 c = new Vec3();
018    private final Vec3 ab = new Vec3();
019    private final Vec3 ap = new Vec3();
020
021    public Triangle() {
022    }
023
024    public Vec3 getPoint(int i) {
025        if(i == 1) {
026            return p2;
027        } else if(i == 2) {
028            return p3;
029        } else {
030            return p1;
031        }
032    }
033
034    public Triangle setTag(int tag) {
035        this.tag = tag;
036
037        return this;
038    }
039
040    public Triangle set(Triangle triangle) {
041        p1.set(triangle.p1);
042        p2.set(triangle.p2);
043        p3.set(triangle.p3);
044        n.set(triangle.n);
045        d = triangle.d;
046        tag = triangle.tag;
047
048        return this;
049    }
050
051    public Triangle calcPlane() {
052        p2.sub(p1, a);
053        p3.sub(p2, b);
054        a.cross(b, n).normalize();
055
056        d = -p1.dot(n);
057
058        return this;
059    }
060
061    public Triangle transform(Mat4 m) {
062        m.transform(p1);
063        m.transform(p2);
064        m.transform(p3);
065
066        return calcPlane();
067    }
068
069    public boolean contains(Vec3 point, float buffer) {
070        for(int i = 0; i != 3; i++) {
071            a.set(getPoint(i));
072            b.set(getPoint(i + 1));
073            b.sub(a, v);
074            n.cross(v, n2).normalize(-buffer);
075            a.add(n2);
076            n.cross(v, n2).normalize();
077
078            float d2 = -a.dot(n2);
079            float s = n2.dot(point) + d2;
080
081            if(s < 0) {
082                return false;
083            }
084        }
085        return true;
086    }
087
088    public boolean intersectsPlane(Vec3 origin, Vec3 direction, float[] time) {
089        float t = direction.dot(n);
090
091        if(Math.abs(t) > 0.0000001) {
092            t = (-d - origin.dot(n)) / t;
093            if(t > 0.0000001 && t < time[0]) {
094                time[0] = t;
095                return true;
096            }
097        }
098        return false;
099    }
100
101    public boolean intersects(Vec3 origin, Vec3 direction, float buffer, float[] time) {
102        float t = time[0];
103
104        if(intersectsPlane(origin, direction, time)) {
105            direction.scale(time[0], p).add(origin);
106            if(contains(p, buffer)) {
107                return true;
108            }
109            time[0] = t;
110        }
111        return false;
112    }
113
114    public Vec3 closestPoint(Vec3 point, Vec3 closestPoint) {
115        float min = Float.MAX_VALUE;
116
117        for(int i = 0; i != 3; i++) {
118            a.set(getPoint(i));
119            b.set(getPoint(i + 1));
120            b.sub(a, ab);
121            point.sub(a, ap);
122
123            float s = ab.dot(ap);
124
125            c.set(a);
126            if(s > 0) {
127                s /= ab.lengthSquared();
128                if(s > 1) {
129                    c.set(b);
130                } else {
131                    a.add(ab.scale(s), c);
132                }
133            }
134            point.sub(c, v);
135
136            float l = v.length();
137
138            if(l < min) {
139                min = l;
140                closestPoint.set(c);
141            }
142        }
143        return closestPoint;
144    }
145
146    @Override
147    public String toString() {
148        return p1 + " : " + p2 + " : " + p3 + " : " + n + " : " + d + " @ " + tag;
149    }
150}
