001package org.game;
002
003import java.util.Vector;
004
005public final class OctTree {
006
007    public static interface Visitor {
008        
009        boolean visit(OctTree tree) throws Exception;
010    }
011
012    public static OctTree create(Vector<Triangle> triangles, int minTrisPerTree) {
013        AABB bounds = new AABB();
014
015        for(Triangle triangle : triangles) {
016            bounds.add(triangle.p1);
017            bounds.add(triangle.p2);
018            bounds.add(triangle.p3);
019        }
020        bounds.buffer(1, 1, 1);
021        
022        return new OctTree(triangles, bounds, minTrisPerTree);
023    }
024    
025    private final AABB bounds = new AABB();
026    private final Vector<OctTree> children = new Vector<>();
027    private final Vector<Triangle> triangles = new Vector<>();
028
029    @SuppressWarnings("unchecked")
030    public OctTree(Vector<Triangle> triangles, AABB bounds, int minTrisPerTree) {
031        this.bounds.set(bounds);
032        if(triangles.size() < minTrisPerTree) {
033            this.triangles.addAll(triangles);
034        } else {
035            Vec3 c = bounds.center(new Vec3());
036            Vec3 l = bounds.min;
037            Vec3 h = bounds.max;
038            AABB[] ba = new AABB[] {
039                new AABB(l.x, l.y, l.z, c.x, c.y, c.z),
040                new AABB(c.x, l.y, l.z, h.x, c.y, c.z),
041                new AABB(c.x, c.y, l.z, h.x, h.y, c.z),
042                new AABB(c.x, c.y, c.z, h.x, h.y, h.z),
043                new AABB(l.x, c.y, l.z, c.x, h.y, c.z),
044                new AABB(l.x, c.y, c.z, c.x, h.y, h.z),
045                new AABB(l.x, l.y, c.z, c.x, c.y, h.z),
046                new AABB(c.x, l.y, c.z, h.x, c.y, h.z)
047            };
048            Vector<Triangle>[] list = new Vector[8];
049            Vector<Triangle> keep = new Vector<>();
050
051            for(int i = 0; i != list.length; i++) {
052                list[i] = new Vector<>();
053            }
054            for(Triangle triangle : triangles) {
055                boolean contained = false;
056
057                for(int i = 0; i != list.length; i++) {
058                    if(
059                        ba[i].contains(triangle.p1) && 
060                        ba[i].contains(triangle.p2) &&
061                        ba[i].contains(triangle.p3)
062                    ) {
063                        contained = true;
064                        list[i].add(triangle);
065                        break;
066                    }
067                }
068                if(!contained) {
069                    keep.add(triangle);
070                }
071            }
072            this.triangles.addAll(keep);
073            for(int i = 0; i != list.length; i++) {
074                if(!list[i].isEmpty()) {
075                    children.add(new OctTree(list[i], ba[i], minTrisPerTree));
076                }
077            }
078        }
079    }
080
081    public AABB getBounds() {
082        return bounds;
083    }
084
085    public int getTriangleCount() {
086        return triangles.size();
087    }
088
089    public Triangle getTriangle(int i, Triangle triangle) {
090        return triangle.set(triangles.get(i));
091    }
092
093    public int getChildCount() {
094        return children.size();
095    }
096
097    public OctTree getChild(int i) {
098        return children.get(i);
099    }
100
101    public void traverse(Visitor v) throws Exception {
102        if(v.visit(this)) {
103            for(OctTree tree : children) {
104                tree.traverse(v);
105            }
106        }
107    }
108}
