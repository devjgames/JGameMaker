005public final class Scene {

007    public final Vec4 backgroundColor = new Vec4(0.15f, 0.15f, 0.15f, 1);
008    public final Vec3 eye = new Vec3(100, 100, 100);
009    public final Vec3 target = new Vec3(0, 0, 0);
010    public final Vec3 up = new Vec3(0, 1, 0);
011    public float fovDegrees = 60;
012    public float zNear = 1;
013    public float zFar = 25000;
014    public final Mat4 projection = new Mat4();

024    private final Mat4 m = new Mat4();
025    private final Vec3 f = new Vec3();
026    private final Vec3 u = new Vec3();
027    private final Vec3 r = new Vec3();
028    private final Vec3 d = new Vec3();
029    private final Vec3 o = new Vec3();
030    private final Vec3 origin = new Vec3();
031    private final Vec3 direction = new Vec3();
032    private final AABB bounds = new AABB();
033    private final boolean inDesign;

063    public void calcBoundsAndTransform() {
064        projection.toIdentity().perspective(fovDegrees, Game.getInstance().aspectRatio(), zNear, zFar);
065        view.toIdentity().lookAt(eye, target, up);
066        root.calcBoundsAndTransform();
067
068        frustum.calcPlanes(projection, view);
069    }
070
071    public void rotateAroundTarget(float dx, float dy) {
072        eye.sub(target, f);
073        m.toIdentity().rotate(dx, 0, 1, 0);
074        m.transformNormal(f.cross(up, r)).normalize();
075        m.transformNormal(f);
076        m.toIdentity().rotate(dy, r);
077        m.transformNormal(r.cross(f, up)).normalize();
078        m.transformNormal(f);
079        target.add(f, eye);
080    }
081
082    public void rotateAroundEye(float dx, float dy) {
083        target.sub(eye, f).normalize();
084        m.toIdentity().rotate(dx, 0, 1, 0);
085        m.transformNormal(f.cross(up, r)).normalize();
086        m.transformNormal(f).normalize();
087        m.toIdentity().rotate(dy, r);
088        m.transformNormal(r.cross(f, up)).normalize();
089        m.transformNormal(f).normalize();
090        eye.add(f, target);
091    }
092
093    public void move(Vec3 point, float dx, float dy) {
094        move(point, dx, dy, null);
095    }
096
097    public void move(Vec3 point, float dx, float dy, Mat4 transform) {
098        float dl = (float)Math.sqrt(dx * dx + dy * dy);
099
100        eye.sub(target, f);
101        d.set(f);
102        f.negate();
103        f.y = 0;
104        if(f.length() > 0.0000001 && dl > 0.0001) {
105            f.normalize().cross(u.set(0, 1, 0), r).normalize();
106            f.scale(dy).add(r.scale(dx));
107            if(transform != null) {
108                transform.transformNormal(f);
109            }
110            point.add(f);
111        }
112        target.add(d, eye);
113    }
114
115    public void move(Vec3 point, float dy) {
116        move(point, dy, null);
117    }
118
119    public void move(Vec3 point, float dy, Mat4 transform) {
120        eye.sub(target, d);
121        u.set(0, dy, 0);
122        if(transform != null) {
123            transform.transformNormal(u);
124        }
125        point.add(u);
126        target.add(d, eye);
127    }
128
129    public void zoom(float amount) {
130        eye.sub(target, d);
131        d.normalize(d.length() + amount);
132        target.add(d, eye);
133    }

136    boolean isectPoint(float cx, float cy, float cz, float halfSize, float[] time) {
137        int w = Game.getInstance().w();
138        int h = Game.getInstance().h();
139        int x = Game.getInstance().mouseX();
140        int y = h - Game.getInstance().mouseY() - 1;
141
142        GFX.unproject(x, y, 0, 0, 0, w, h, projection, view, origin);
143        GFX.unproject(x, y, 1, 0, 0, w, h, projection, view, direction);
144
145        direction.sub(origin).normalize();
146
147        bounds.min.set(cx, cy, cz).sub(halfSize, halfSize, halfSize);
148        bounds.max.set(cx, cy, cz).add(halfSize, halfSize, halfSize);
149
150        if(bounds.isects(origin, direction, time)) {
151            return true;
152        }
153        return false;
154    }
155
156    static Scene next(Scene scene) {
157        File f = scene.getLoadFile();
158        Game game = Game.getInstance();
159
160        if(f != null) {
161            if(f.exists()) {
162                try {
163                    scene = null;
164                    game.getAssets().clear();
165                    scene = SceneSerializer.deserialize(false, f);
166                } catch(Exception ex) {
167                    Log.put(0, ex);
168                }
169            }
170        }
171        return scene;
172    }
