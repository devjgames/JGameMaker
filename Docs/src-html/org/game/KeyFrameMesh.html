017        @Override
018        public Object load(File file, AssetManager assets) throws Exception {
019            String[] lines = new String(IO.readAllBytes(file)).split("\\n+");
020            File meshFile = null;
021            File directory = file.getParentFile();
022            File textureFile = null;
023
024            for(String line : lines) {
025                String tLine = line.trim();
026
027                if(tLine.startsWith("texture ")) {
028                    textureFile = IO.file(tLine.substring(7).trim());
029                } else if(tLine.startsWith("mesh ")) {
030                    meshFile = IO.file(directory, tLine.substring(4).trim());
031                }
032            }
033
034            KeyFrameMesh mesh = new KeyFrameMesh(meshFile, file, textureFile);
035
036            return mesh;
037        }
038        
039    }

060    KeyFrameMesh(File file, File kfmFile, File textureFile) throws IOException {
061        BinReader reader = new BinReader(IO.readAllBytes(file));
062
063        this.file = IO.file(kfmFile.getName());
064
065        File decalFile = IO.file(AssetManager.getRoot(), IO.getFilenameWithoutExtension(textureFile) + "_DECAL.png");
066
067        if(decalFile.exists()) {
068            try {
069                decal = Game.getInstance().getAssets().load(IO.file(decalFile.getName()));
070            } catch(Exception ex) {
071                Log.put(0, ex);
072            }
073        }
074
075        if(textureFile != null) {
076            try {
077                texture = Game.getInstance().getAssets().load(textureFile);
078            } catch(Exception ex) {
079                Log.put(0, ex);
080            }
081        }
082
083        header = new Header(reader);
084        reader.setPosition(header.offST);
085        textureCoordinates = new TextureCoordinate[header.numST];
086        for(int i = 0; i != header.numST; i++) {
087            textureCoordinates[i] = new TextureCoordinate(reader);
088        }
089        reader.setPosition(header.offTris);
090        triangles = new TriangleIndices[header.numTris];
091        for(int i = 0; i != header. numTris; i++) {
092            triangles[i] = new TriangleIndices(reader);
093        }
094        frames = new Frame[header.numFrames];
095        for(int i = 0; i != header.numFrames; i++) {
096            reader.setPosition(header.offFrames + header.frameSize * i);
097            frames[i] = new Frame(reader, header);
098        }
099
100        reset();
101
102        bounds.set(frames[0].bounds);
103
104        normals = MD2Normals.cloneNormals();
105    }
106
107    KeyFrameMesh(KeyFrameMesh mesh) {
108        this.file = mesh.file;
109    
110        header = mesh.header;
111        textureCoordinates = mesh.textureCoordinates;
112        triangles = mesh.triangles;
113        frames = mesh.frames;
114
115        if(mesh.texture != null) {
116            try {
117                texture = mesh.texture.newInstance();
118            } catch(Exception ex) {
119                Log.put(0, ex);
120            }
121        }
122
123        if(mesh.decal != null) {
124            try {
125                decal = mesh.decal.newInstance();
126            } catch(Exception ex) {
127                Log.put(0, ex);
128            }
129        }
130        reset();
131
132        bounds.set(frames[0].bounds);
133
134        normals = mesh.normals;
135    }

172    public void setSequence(int start, int end, int speed, boolean looping) {
173        if(start >= 0 && start < frames.length && end >= 0 && end < frames.length && start <= end) {
174            if(start != this.start || end != this.end || speed != this.speed || looping != this.looping) {
175                this.start = start;
176                this.end = end;
177                this.speed = speed;
178                this.looping = looping;
179                reset();
180            }
181        }
182    }

227    @Override
228    public Triangle getTriangle(Scene scene, SceneNode node, int i, Triangle triangle) {
229        int f1 = frame;
230        int f2 = frame + 1;
231
232        if(f1 == end) {
233            f2 = start;
234        }
235
236        TriangleVertex f1v1 = frames[f1].vertices[triangles[i].vertex[0]];
237        TriangleVertex f1v2 = frames[f1].vertices[triangles[i].vertex[1]];
238        TriangleVertex f1v3 = frames[f1].vertices[triangles[i].vertex[2]];
239        TriangleVertex f2v1 = frames[f2].vertices[triangles[i].vertex[0]];
240        TriangleVertex f2v2 = frames[f2].vertices[triangles[i].vertex[1]];
241        TriangleVertex f2v3 = frames[f2].vertices[triangles[i].vertex[2]];
242
243        float f1x1 = f1v1.v[0] * frames[f1].scale.x + frames[f1].translate.x;
244        float f1y1 = f1v1.v[1] * frames[f1].scale.y + frames[f1].translate.y;
245        float f1z1 = f1v1.v[2] * frames[f1].scale.z + frames[f1].translate.z;
246        float f1x2 = f1v2.v[0] * frames[f1].scale.x + frames[f1].translate.x;
247        float f1y2 = f1v2.v[1] * frames[f1].scale.y + frames[f1].translate.y;
248        float f1z2 = f1v2.v[2] * frames[f1].scale.z + frames[f1].translate.z;
249        float f1x3 = f1v3.v[0] * frames[f1].scale.x + frames[f1].translate.x;
250        float f1y3 = f1v3.v[1] * frames[f1].scale.y + frames[f1].translate.y;
251        float f1z3 = f1v3.v[2] * frames[f1].scale.z + frames[f1].translate.z;
252        float f2x1 = f2v1.v[0] * frames[f2].scale.x + frames[f2].translate.x;
253        float f2y1 = f2v1.v[1] * frames[f2].scale.y + frames[f2].translate.y;
254        float f2z1 = f2v1.v[2] * frames[f2].scale.z + frames[f2].translate.z;
255        float f2x2 = f2v2.v[0] * frames[f2].scale.x + frames[f2].translate.x;
256        float f2y2 = f2v2.v[1] * frames[f2].scale.y + frames[f2].translate.y;
257        float f2z2 = f2v2.v[2] * frames[f2].scale.z + frames[f2].translate.z;
258        float f2x3 = f2v3.v[0] * frames[f2].scale.x + frames[f2].translate.x;
259        float f2y3 = f2v3.v[1] * frames[f2].scale.y + frames[f2].translate.y;
260        float f2z3 = f2v3.v[2] * frames[f2].scale.z + frames[f2].translate.z;
261
262        triangle.p1.x = f1x1 + amount * (f2x1 - f1x1);
263        triangle.p1.y = f1y1 + amount * (f2y1 - f1y1);
264        triangle.p1.z = f1z1 + amount * (f2z1 - f1z1);
265
266        triangle.p2.x = f1x2 + amount * (f2x2 - f1x2);
267        triangle.p2.y = f1y2 + amount * (f2y2 - f1y2);
268        triangle.p2.z = f1z2 + amount * (f2z2 - f1z2);
269
270        triangle.p3.x = f1x3 + amount * (f2x3 - f1x3);
271        triangle.p3.y = f1y3 + amount * (f2y3 - f1y3);
272        triangle.p3.z = f1z3 + amount * (f2z3 - f1z3);
273
274        return triangle.calcPlane();
275    }

277    @Override
278    public void update(Scene scene, SceneNode node) throws Exception {
279        if(done) {
280            return;
281        }
282
283        amount += speed * Game.getInstance().elapsedTime();
284        if(amount >= 1) {
285            if(looping) {
286                if(frame == end) {
287                    frame = start;
288                } else {
289                    frame++;
290                }
291                amount = 0;
292            } else if(frame == end - 1) {
293                amount = 1;
294                done = true;
295            } else {
296                frame++;
297                amount = 0;
298            }
299        }
300
301        int f1 = frame;
302        int f2 = frame + 1;
303
304        if(f1 == end) {
305            f2 = start;
306        }
307
308        AABB b1 = frames[f1].bounds;
309        AABB b2 = frames[f2].bounds;
310
311        b1.min.lerp(b2.min, amount, bounds.min);
312        b1.max.lerp(b2.max, amount, bounds.max);
313    }

315    @Override
316    public int render(Scene scene, SceneNode node, Vector<SceneNode> lights) throws Exception {
317        LightRenderer renderer = Game.getInstance().getRenderer(LightRenderer.class);
318        int i1 = frame;
319        int i2 = frame + 1;
320
321        if(i1 == end) {
322            i2 = start;
323        }
324
325        Frame f1 = frames[i1];
326        Frame f2 = frames[i2];
327
328        renderer.begin(scene, node, lights, texture, decal);
329
330        for(int i = 0; i != header.numTris; i++) {
331            for(int j = 2; j != -1; j--) {
332                TriangleVertex v1 = f1.vertices[triangles[i].vertex[j]];
333                TriangleVertex v2 = f2.vertices[triangles[i].vertex[j]];
334                TextureCoordinate coord = textureCoordinates[triangles[i].st[j]];
335                float x1 = v1.v[0] * f1.scale.x + f1.translate.x;
336                float y1 = v1.v[1] * f1.scale.y + f1.translate.y;
337                float z1 = v1.v[2] * f1.scale.z + f1.translate.z;
338                float x2 = v2.v[0] * f2.scale.x + f2.translate.x;
339                float y2 = v2.v[1] * f2.scale.y + f2.translate.y;
340                float z2 = v2.v[2] * f2.scale.z + f2.translate.z;
341
342                vertex.position.x = x1 + amount * (x2 - x1);
343                vertex.position.y = y1 + amount * (y2 - y1);
344                vertex.position.z = z1 + amount * (z2 - z1);
345                vertex.textureCoordinate.x = coord.s / (float)header.skinWidth;
346                vertex.textureCoordinate.y = coord.t / (float)header.skinHeight;
347                vertex.normal.x = normals[v1.n][0] + amount * (normals[v2.n][0] - normals[v1.n][0]);
348                vertex.normal.y = normals[v1.n][1] + amount * (normals[v2.n][1] - normals[v1.n][1]);
349                vertex.normal.z = normals[v1.n][2] + amount * (normals[v2.n][2] - normals[v1.n][2]);
350
351                renderer.push(vertex);
352            }
353        }
354        renderer.end();
355
356        return getTriangleCount();
357    }
