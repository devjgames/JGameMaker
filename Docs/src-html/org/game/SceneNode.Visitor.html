066    public SceneNode(Scene scene, SceneNode node) throws Exception {
067        if(node.renderable != null) {
068            renderable = node.renderable.newInstance();
069        }
070        Utils.copy(node, this);
071        for(SceneNode child : node) {
072            addChild(new SceneNode(scene, child));
073        }
074        for(SceneNodeComponent component : node.components) {
075            component.newInstance(scene, this);
076        }
077    }

115    public OctTree getOctTree(Scene scene) {
116        if(octTree == null && !dynamic && renderable != null) {
117            Vector<Triangle> triangles = new Vector<>();
118
119            for(int i = 0; i != getTriangleCount(); i++) {
120                triangles.add(getTriangle(scene, i, new Triangle()));
121            }
122            octTree = OctTree.create(triangles, minTrisPerTree);
123        }
124        return octTree;
125    }

171    public void calcBoundsAndTransform() {
172        localModel
173            .toIdentity()
174            .translate(position)
175            .mul(
176                m.set(
177                    r.x, u.x, f.x, 0,
178                    r.y, u.y, f.y, 0,
179                    r.z, u.z, f.z, 0,
180                    0, 0, 0, 1
181                )
182            )
183            .scale(scale);
184        model.set(localModel);
185        if(parent != null) {
186            model.set(parent.model).mul(localModel);
187        }
188        modelIT.set(model).invert().transpose();
189        model.transform(absolutePosition.toZero());
190        bounds.clear();
191        if(renderable != null) {
192            bounds.set(renderable.getBounds()).transform(model);
193        } 
194        if(emitsLight) {
195            Vec3 p = absolutePosition;
196            float r = lightRadius;
197
198            bounds.add(p.x - r, p.y - r, p.z - r);
199            bounds.add(p.x + r, p.y + r, p.z + r);
200        }
201        for(SceneNode node : this) {
202            node.calcBoundsAndTransform();
203            bounds.add(node.bounds);
204        }
205    }

207    public void rotate(int axis, float degrees) {
208        if(axis == 0) {
209            m.toIdentity().rotate(degrees, r);
210            m.transformNormal(u).normalize();
211            m.transformNormal(f).normalize();
212        } else if(axis == 1) {
213            m.toIdentity().rotate(degrees, u);
214            m.transformNormal(r).normalize();
215            m.transformNormal(f).normalize();
216        } else {
217            m.toIdentity().rotate(degrees, f);
218            m.transformNormal(r).normalize();
219            m.transformNormal(u).normalize();
220        }
221    }

223    public boolean move(Vec3 p1, Vec3 p2, float speed) {
224        float dx = p2.x - p1.x;
225        float dy = p2.y - p1.y;
226        float dz = p2.z - p1.z;
227        float len = (float)Math.sqrt(dx * dx + dy * dy + dz * dz);
228        float nx = dx / len;
229        float ny = dy / len;
230        float nz = dz / len;
231        float vl = Game.getInstance().elapsedTime() * speed;
232
233        position.add(nx * vl, ny * vl, nz * vl);
234
235        float vx = position.x - p1.x;
236        float vy = position.y - p1.y;
237        float vz = position.z - p1.z;
238        float s = vx * nx + vy * ny + vz * nz;
239
240        r.set(nx, ny, nz);
241        u.set(0, 1, 0);
242        r.cross(u, f).normalize();
243        f.cross(r, u).normalize();
244
245        if(s >= len) {
246            position.set(p2);
247            return true;
248        }
249        return false;
250    }

252    public void traverse(Visitor v) throws Exception {
253        if(v.visit(this)) {
254            for(SceneNode node : this) {
255                node.traverse(v);
256            }
257        }
258    }
259
260    public SceneNode find(Visitor v) throws Exception {
261        if(v.visit(this)) {
262            return this;
263        }
264        for(SceneNode node : this) {
265            SceneNode r = node.find(v);
266
267            if(r != null) {
268                return r;
269            }
270        }
271        return null;
272    }

274    void addBrushes(Scene scene, JComboBox<SceneNode> brushes) throws Exception {
275        if(isBrush) {
276            brushes.addItem(new SceneNode(scene, this));
277        } else {

279                node.addBrushes(scene, brushes);

284    void init() {
285        for(SceneNodeComponent component : components) {
286            if(!component.isComplete) {
287                try {
288                    component.init();
289                } catch(Exception ex) {
290                    Log.put(0, ex);
291                }
292            }
293        }
294    }
295
296    void start() {
297        for(SceneNodeComponent component : components) {
298            if(!component.isComplete) {
299                try {
300                    component.start();
301                } catch(Exception ex) {
302                    Log.put(0, ex);
303                } finally {
304                    component.isComplete = true;
305                }
306            }
307        }
308    }
309
310    void update(Scene scene) {
311        for(SceneNodeComponent component : components) {
312            try {
313                component.update();
314            } catch(Exception ex) {
315                Log.put(0, ex);
316            }
317        }
318        if(renderable != null) {
319            try {
320                renderable.update(scene, this);
321            } catch(Exception ex) {
322                Log.put(0, ex);
323            }
324        }
325    }

327    void renderSprites(SpriteRenderer renderer) {
328        for(SceneNodeComponent component : components) {
329            try {
330                component.renderSprites(renderer); 
331            } catch(Exception ex) {
332                Log.put(0, ex);
333            }
334        }
335    }

337    void reloadComponents(Scene scene) {
338        Vector<SceneNodeComponent> old = new Vector<>(components);
339
340        components.clear();
341        for(int i = 0; i != old.size(); i++) {
342            old.get(i).newInstance(scene, this);
343        }
344        for(SceneNode node : children) {
345            node.reloadComponents(scene);
346        }
347    }
