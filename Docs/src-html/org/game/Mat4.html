055    public Mat4 mul(
056        float m00, float m01, float m02, float m03,
057        float m10, float m11, float m12, float m13,
058        float m20, float m21, float m22, float m23,
059        float m30, float m31, float m32, float m33
060    ) {
061        return set( 
062            this.m00 * m00 + this.m01 * m10 + this.m02 * m20 + this.m03 * m30,
063            this.m00 * m01 + this.m01 * m11 + this.m02 * m21 + this.m03 * m31,
064            this.m00 * m02 + this.m01 * m12 + this.m02 * m22 + this.m03 * m32,
065            this.m00 * m03 + this.m01 * m13 + this.m02 * m23 + this.m03 * m33,
066
067            this.m10 * m00 + this.m11 * m10 + this.m12 * m20 + this.m13 * m30,
068            this.m10 * m01 + this.m11 * m11 + this.m12 * m21 + this.m13 * m31,
069            this.m10 * m02 + this.m11 * m12 + this.m12 * m22 + this.m13 * m32,
070            this.m10 * m03 + this.m11 * m13 + this.m12 * m23 + this.m13 * m33,
071
072            this.m20 * m00 + this.m21 * m10 + this.m22 * m20 + this.m23 * m30,
073            this.m20 * m01 + this.m21 * m11 + this.m22 * m21 + this.m23 * m31,
074            this.m20 * m02 + this.m21 * m12 + this.m22 * m22 + this.m23 * m32,
075            this.m20 * m03 + this.m21 * m13 + this.m22 * m23 + this.m23 * m33,
076
077            this.m30 * m00 + this.m31 * m10 + this.m32 * m20 + this.m33 * m30,
078            this.m30 * m01 + this.m31 * m11 + this.m32 * m21 + this.m33 * m31,
079            this.m30 * m02 + this.m31 * m12 + this.m32 * m22 + this.m33 * m32,
080            this.m30 * m03 + this.m31 * m13 + this.m32 * m23 + this.m33 * m33
081        );

137    private float fma(float a, float b, float c) {
138        return a * b + c;
139    }

141    public Mat4 invert() {
142        float a = m00 * m11 - m01 * m10;
143        float b = m00 * m12 - m02 * m10;
144        float c = m00 * m13 - m03 * m10;
145        float d = m01 * m12 - m02 * m11;
146        float e = m01 * m13 - m03 * m11;
147        float f = m02 * m13 - m03 * m12;
148        float g = m20 * m31 - m21 * m30;
149        float h = m20 * m32 - m22 * m30;
150        float i = m20 * m33 - m23 * m30;
151        float j = m21 * m32 - m22 * m31;
152        float k = m21 * m33 - m23 * m31;
153        float l = m22 * m33 - m23 * m32;
154        float det = a * l - b * k + c * j + d * i - e * h + f * g;
155        det = 1.0f / det;
156        float nm00 = fma( m11, l, fma(-m12, k,  m13 * j)) * det;
157        float nm01 = fma(-m01, l, fma( m02, k, -m03 * j)) * det;
158        float nm02 = fma( m31, f, fma(-m32, e,  m33 * d)) * det;
159        float nm03 = fma(-m21, f, fma( m22, e, -m23 * d)) * det;
160        float nm10 = fma(-m10, l, fma( m12, i, -m13 * h)) * det;
161        float nm11 = fma( m00, l, fma(-m02, i,  m03 * h)) * det;
162        float nm12 = fma(-m30, f, fma( m32, c, -m33 * b)) * det;
163        float nm13 = fma( m20, f, fma(-m22, c,  m23 * b)) * det;
164        float nm20 = fma( m10, k, fma(-m11, i,  m13 * g)) * det;
165        float nm21 = fma(-m00, k, fma( m01, i, -m03 * g)) * det;
166        float nm22 = fma( m30, e, fma(-m31, c,  m33 * a)) * det;
167        float nm23 = fma(-m20, e, fma( m21, c, -m23 * a)) * det;
168        float nm30 = fma(-m10, j, fma( m11, h, -m12 * g)) * det;
169        float nm31 = fma( m00, j, fma(-m01, h,  m02 * g)) * det;
170        float nm32 = fma(-m30, d, fma( m31, b, -m32 * a)) * det;
171        float nm33 = fma( m20, d, fma(-m21, b,  m22 * a)) * det;
172
173        return set( 
174            nm00, nm01, nm02, nm03,
175            nm10, nm11, nm12, nm13,
176            nm20, nm21, nm22, nm23,
177            nm30, nm31, nm32, nm33
178        );
179    }

207    public Mat4 rotate(float degrees, float x, float y, float z) {
208        float radians = (float)Math.toRadians(degrees);
209        float c = (float)Math.cos(radians);
210        float s = (float)Math.sin(radians);
211        float l = (float)Math.sqrt(x * x + y * y + z * z);
212
213        x /= l;
214        y /= l;
215        z /= l;
216        return mul( 
217            x * x * (1 - c) + c, x * y * (1 - c) - z * s, x * z * (1 - c) + y * s, 0,
218            y * x * (1 - c) + z * s, y * y * (1 - c) + c, y * z * (1 - c) - x * s, 0,
219            x * z * (1 - c) - y * s, y * z * (1 - c) + x * s, z * z * (1 - c) + c, 0,
220            0, 0, 0, 1
221        );
222    }

228    public Mat4 ortho(float left, float right, float bottom, float top, float zNear, float zFar) {
229        float sx = 2 / (right - left);
230        float sy = 2 / (top - bottom);
231        float sz = -2 / (zFar - zNear);
232        float tx = -(right + left) / (right - left);
233        float ty = -(top + bottom) / (top - bottom);
234        float tz = -(zFar + zNear) / (zFar - zNear);
235
236        return mul( 
237            sx, 0, 0, tx,
238            0, sy, 0, ty,
239            0, 0, sz, tz,
240            0, 0, 0, 1
241        );
242    }

244    public Mat4 perspective(float fovDegrees, float aspectRatio, float zNear, float zFar) {
245        float radians = (float)Math.toRadians(fovDegrees);
246        float f = 1 / (float)Math.tan(radians / 2);
247        return mul( 
248            f / aspectRatio, 0, 0, 0,
249            0, f, 0, 0,
250            0, 0, (zFar + zNear) / (zNear - zFar), 2 * zFar * zNear / (zNear - zFar),
251            0, 0, -1, 0
252        );
253    }

255    public Mat4 lookAt(float ex, float ey, float ez, float tx, float ty, float tz, float ux, float uy, float uz) {
256        float fx = tx - ex;
257        float fy = ty - ey;
258        float fz = tz - ez;
259        float fl = (float)Math.sqrt(fx * fx + fy * fy + fz * fz);
260        float ul = (float)Math.sqrt(ux * ux + uy * uy + uz * uz);
261
262        fx /= fl;
263        fy /= fl;
264        fz /= fl;
265        ux /= ul;
266        uy /= ul;
267        uz /= ul;
268
269        float rx = fy * uz - fz * uy;
270        float ry = fz * ux - fx * uz;
271        float rz = fx * uy - fy * ux;
272        float rl = (float)Math.sqrt(rx * rx + ry * ry + rz * rz);
273
274        rx /= rl;
275        ry /= rl;
276        rz /= rl;
277
278        ux = ry * fz - rz * fy;
279        uy = rz * fx - rx * fz;
280        uz = rx * fy - ry * fx;
281        ul = (float)Math.sqrt(ux * ux + uy * uy + uz * uz);
282
283        fx = -fx;
284        fy = -fy;
285        fz = -fz;
286
287        return mul(
288            rx, ry, rz, -(ex * rx + ey * ry + ez * rz),
289            ux, uy, uz, -(ex * ux + ey * uy + ez * uz),
290            fx, fy, fz, -(ex * fx + ey * fy + ez * fz), 
291            0, 0, 0, 1
292        );
293    }

299    public Vec4 transform(Vec4 v) {
300        float tx = m00 * v.x + m01 * v.y + m02 * v.z + m03 * v.w;
301        float ty = m10 * v.x + m11 * v.y + m12 * v.z + m13 * v.w;
302        float tz = m20 * v.x + m21 * v.y + m22 * v.z + m23 * v.w;
303        float tw = m30 * v.x + m31 * v.y + m32 * v.z + m33 * v.w;
304
305        return v.set(tx, ty, tz, tw);
306    }
