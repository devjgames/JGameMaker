001package org.game;
002
003import java.io.ByteArrayOutputStream;
004import java.io.File;
005import java.io.FileInputStream;
006import java.io.FileOutputStream;
007import java.io.IOException;
008import java.io.InputStream;
009
010public final class IO {
011    
012    public static byte[] readAllBytes(InputStream input) throws IOException {
013        ByteArrayOutputStream output = null;
014        byte[] buf = new byte[1024];
015        int n;
016
017        try {
018            output = new ByteArrayOutputStream(1024);
019            while((n = input.read(buf)) >= 0) {
020                if(n > 0) {
021                    output.write(buf, 0, n);
022                }
023            }
024            return output.toByteArray();
025        } finally {
026            if(output != null) {
027                output.close();
028            }
029        }
030    }
031
032    public static byte[] readAllBytes(File file) throws IOException {
033        FileInputStream input = null;
034
035        try {
036            return readAllBytes(input = new FileInputStream(file));
037        } finally {
038            if(input != null) {
039                input.close();
040            }
041        }
042    }
043
044    public static byte[] readAllBytes(Class<?> cls, String name) throws IOException {
045        InputStream input = null;
046
047        try {
048            return readAllBytes(input = cls.getResourceAsStream(name));
049        } finally {
050            if(input != null) {
051                input.close();
052            }
053        }
054    }
055
056    public static void writeAllBytes(byte[] bytes, File file) throws IOException {
057        FileOutputStream output = null;
058
059        try {
060            output = new FileOutputStream(file);
061            output.write(bytes);
062        } finally {
063            if(output != null) {
064                output.close();
065            }
066        }
067    }
068
069    public static void appendAllBytes(byte[] bytes, File file) throws IOException {
070        if(!file.exists()) {
071            writeAllBytes(bytes, file);
072        } else {
073            FileOutputStream output = null;
074            byte[] b = readAllBytes(file);
075
076            try {
077                output = new FileOutputStream(file);
078                output.write(b);
079                output.write(bytes);
080            } finally {
081                if(output != null) {
082                    output.close();
083                }
084            }
085        }
086    }
087
088    public static File file(String path) {
089        return new File(path.replace('\\', File.separatorChar).replace('/', File.separatorChar));
090    }
091
092    public static File file(File file, String path) {
093        return new File(file, IO.file(path).getPath());
094    }
095
096    public static String getFilenameWithoutExtension(File file) {
097        String name = file.getName();
098        int i = name.lastIndexOf('.');
099
100        if(i >= 0) {
101            name = name.substring(0, i);
102        }
103        return name;
104    }
105
106    public static String getExtension(File file) {
107        String name = file.getName();
108        String extension = "";
109        int i = name.lastIndexOf('.');
110
111        if(i >= 0) {
112            extension = name.substring(i);
113        }
114        return extension;
115    }
116}
