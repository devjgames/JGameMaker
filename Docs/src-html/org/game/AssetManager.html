001package org.game;
002
003import java.io.File;
004import java.util.Hashtable;
005
006public final class AssetManager extends Resource {
007    
008    private static Hashtable<String, AssetLoader> assetLoaders = new Hashtable<>();
009    
010    static File root = null;
011
012    static void registerAssetLoader(String extension, AssetLoader assetLoader) {
013        assetLoaders.put(extension, assetLoader);
014    }
015
016    public static File getRoot() {
017        return root;
018    }
019
020    public final ResourceManager resources = new ResourceManager();
021    
022    private Hashtable<String, Object> assets = new Hashtable<>();
023
024    AssetManager() {
025        registerAssetLoader(".png", new Texture.Loader());
026        registerAssetLoader(".wav", new SoundLoader());
027        registerAssetLoader(".kfm", new KeyFrameMesh.KeyFrameMeshLoader());
028        registerAssetLoader(".obj", new MeshLoader());
029    }
030
031    @SuppressWarnings("unchecked")
032    public <T extends Object> T load(File file) throws Exception {
033        String key = file.getPath();
034
035        if(!assets.containsKey(key)) {
036            Log.put(1, "loading asset '" + key + "' ...");
037            assets.put(key, assetLoaders.get(IO.getExtension(file)).load(IO.file(root, key), this));
038        }
039        return (T)assets.get(key);
040    }
041
042    void unload(File file) throws Exception {
043        String key = file.getPath();
044
045        if(assets.containsKey(key)) {
046            Object asset = assets.get(key);
047
048            if(asset instanceof Resource) {
049                ((Resource)asset).destroy();
050            }
051            assets.remove(key);
052        }
053    }
054
055    void clear() throws Exception {
056        resources.clear();
057
058        for(String key : assets.keySet()) {
059            Object asset = assets.get(key);
060
061            if(asset instanceof Resource) {
062                ((Resource)asset).destroy();
063            }
064        }
065        assets.clear();
066    }
067
068    @Override
069    void destroy() throws Exception {
070        clear();
071        resources.destroy();
072        super.destroy();
073    }
074}
