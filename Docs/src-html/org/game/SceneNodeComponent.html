001package org.game;
002
003import java.io.File;
004import java.util.Hashtable;
005import java.util.Vector;
006
007import javax.script.Bindings;
008import javax.script.Invocable;
009import javax.script.ScriptContext;
010import javax.script.ScriptEngine;
011
012public final class SceneNodeComponent {
013    
014    public final Hashtable<String, Object> properties = new Hashtable<>();
015    public final Vector<String> propertyNames = new Vector<>();
016
017    private SceneNode node = null;
018    private Scene scene = null;
019    private ScriptEngine engine;
020    private Bindings bindings;
021    private final File jsFile;
022    private final String type;
023
024    boolean isComplete = false;
025
026    SceneNodeComponent(ScriptEngine engine, File jsFile) {
027        this.engine = engine;
028        this.bindings = engine.getBindings(ScriptContext.ENGINE_SCOPE);
029        this.jsFile = jsFile;
030        this.type = IO.getFilenameWithoutExtension(jsFile);
031
032        try {
033            Invocable invocable = (Invocable)engine;
034
035            if(functionExists("create")) {
036                invocable.invokeFunction("create", this);
037            }
038        } catch(Exception ex) {
039            Log.put(0, ex);
040            engine = null;
041        }
042    }
043
044    public SceneNode node() {
045        return node;
046    }
047
048    public Scene scene() {
049        return scene;
050    }
051
052    public File getJSFile() {
053        return jsFile;
054    }
055
056    public String getType() {
057        return type;
058    }
059
060    void init() {
061        if(engine != null) {
062            try {
063                Invocable invocable = (Invocable)engine;
064
065                if(functionExists("init")) {
066                    invocable.invokeFunction("init", this);
067                }
068            } catch(Exception ex) {
069                Log.put(0, ex);
070                engine = null;
071            }
072        }
073    }
074
075    void start() {
076        if(engine != null) {
077            try {
078                Invocable invocable = (Invocable)engine;
079
080                if(functionExists("start")) {
081                    invocable.invokeFunction("start", this);
082                }
083            } catch(Exception ex) {
084                Log.put(0, ex);
085                engine = null;
086            }
087        }
088    }
089
090    void update() {
091        if(engine != null) {
092            try {
093                Invocable invocable = (Invocable)engine;
094
095                if(functionExists("update")) {
096                    invocable.invokeFunction("update", this);
097                }
098            } catch(Exception ex) {
099                Log.put(0, ex);
100                engine = null;
101            }
102        }
103    }
104
105    void renderSprites(SpriteRenderer renderer) {
106        if(engine != null) {
107            try {
108                Invocable invocable = (Invocable)engine;
109
110                if(functionExists("renderSprites")) {
111                    invocable.invokeFunction("renderSprites", this, renderer);
112                }
113            } catch(Exception ex) {
114                Log.put(0, ex);
115                engine = null;
116            }
117        }
118    }
119
120    public void sendMessage(SceneNodeComponent component, String type) {
121        if(engine != null) {
122            try {
123                Invocable invocable = (Invocable)engine;
124
125                if(functionExists("receiveMessage")) {
126                    invocable.invokeFunction("receiveMessage", this, component, type);
127                }
128            } catch(Exception ex) {
129                Log.put(0, ex);
130                engine = null;
131            }
132        }
133    }
134
135    String loadSceneName() {
136        if(engine != null) {
137            try {
138                Invocable invocable = (Invocable)engine;
139
140                if(functionExists("loadSceneName")) {
141                    Object r = invocable.invokeFunction("loadSceneName", this);
142
143                    if(r instanceof String) {
144                        return (String)r;
145                    }
146                }
147            } catch(Exception ex) {
148                Log.put(0, ex);
149                engine = null;
150            }
151        }
152        return null;
153    }
154
155    SceneNodeComponent newInstance(Scene scene, SceneNode node) {
156        try {
157            SceneNodeComponent component = new SceneNodeComponent(engine, jsFile);
158
159            component.node = node;
160            component.scene = scene;
161
162            node.addComponent(component);
163
164            for(String key : properties.keySet()) {
165                Object value = properties.get(key);
166                Class<?> cls = value.getClass();
167
168                if(
169                    int.class.isAssignableFrom(cls) ||
170                    float.class.isAssignableFrom(cls) ||
171                    double.class.isAssignableFrom(cls) ||
172                    boolean.class.isAssignableFrom(cls) ||
173                    Boolean.class.isAssignableFrom(cls) ||
174                    Integer.class.isAssignableFrom(cls) ||
175                    Float.class.isAssignableFrom(cls) ||
176                    Double.class.isAssignableFrom(cls) ||
177                    String.class.isAssignableFrom(cls) || 
178                    cls.isEnum()
179                ) {
180                    component.properties.put(key, value);
181                } else if(Vec2.class.isAssignableFrom(cls)) {
182                    component.properties.put(key, new Vec2((Vec2)value));
183                } else if(Vec3.class.isAssignableFrom(cls)) {
184                    component.properties.put(key, new Vec3((Vec3)value));
185                } else if(Vec4.class.isAssignableFrom(cls)) {
186                    component.properties.put(key, new Vec4((Vec4)value));
187                } else if(UIEnum.class.isAssignableFrom(cls)) {
188                    UIEnum uiEnum = (UIEnum)component.properties.get(key);
189
190                    uiEnum.setValue(((UIEnum)value).getValue());
191                }
192            }
193            return component;
194        } catch(Exception ex) {
195            Log.put(0, ex);
196        }
197        return null;
198    }
199
200    private boolean functionExists(String name) {
201        if(bindings.containsKey(name)) {
202            return true;
203        }
204        return false;
205    }
206
207    @Override
208    public String toString() {
209        return type;
210    }
211}
