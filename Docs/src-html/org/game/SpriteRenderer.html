001package org.game;
002
003
004import java.nio.FloatBuffer;
005
006import org.lwjgl.BufferUtils;
007import org.lwjgl.opengl.GL11;
008import org.lwjgl.opengl.GL15;
009import org.lwjgl.opengl.GL20;
010
011
012public final class SpriteRenderer extends Renderer {
013
014    public static final int VERTEX_STRIDE = 8;
015    
016    private final Shader shader;
017    private final int vbo;
018    private FloatBuffer vBuf = BufferUtils.createFloatBuffer(6 * VERTEX_STRIDE);
019    private Texture texture = null;
020    private final Mat4 matrix = new Mat4();
021
022    public SpriteRenderer() throws Exception {
023        shader = new Shader(
024            IO.readAllBytes(Shader.class, "/org/game/glsl/SpriteVertexShader.glsl"),
025            IO.readAllBytes(Shader.class, "/org/game/glsl/SpriteFragmentShader.glsl"),
026            "aPosition", "aTextureCoordinate", "aColor"
027        );
028        
029
030        vbo = GL15.glGenBuffers();
031    }
032
033    void begin() {
034        GL15.glBindBuffer(GL15.GL_ARRAY_BUFFER, vbo);
035        shader.begin();
036        shader.set("uProjection", matrix.toIdentity().ortho(0, Game.getInstance().w(), Game.getInstance().h(), 0, -1, 1));
037        GL20.glVertexAttribPointer(0, 2, GL11.GL_FLOAT, false, VERTEX_STRIDE * 4, 0);
038        GL20.glVertexAttribPointer(1, 2, GL11.GL_FLOAT, false, VERTEX_STRIDE * 4, 4 * 2);
039        GL20.glVertexAttribPointer(2, 4, GL11.GL_FLOAT, false, VERTEX_STRIDE * 4, 4 * 4);
040
041        texture = null;
042
043        GFX.setDepthState(DepthState.NONE);
044        GFX.setBlendState(BlendState.ALPHA);
045        GFX.setCullState(CullState.NONE);
046    }
047
048    public void beginSprite(Texture texture) {
049        shader.bind(GL11.GL_TEXTURE_2D, "uTexture", 0, (this.texture = texture).id);
050
051        vBuf.limit(vBuf.capacity());
052        vBuf.position(0);
053    }
054
055    public void push(int sx, int sy, int sw, int sh, int dx, int dy, int dw, int dh, float r, float g, float b, float a, boolean flip) {
056        if(texture == null ){
057            return;
058        }
059
060        float tw = texture.w;
061        float th = texture.h;
062        float sx1 = sx / tw;
063        float sy1 = sy / th;
064        float sx2 = (sx + sw) / tw;
065        float sy2 = (sy + sh) / th;
066        float dx1 = dx;
067        float dy1 = dy;
068        float dx2 = dx + dw;
069        float dy2 = dy + dh;
070
071        if(vBuf.position() + 6 * VERTEX_STRIDE > vBuf.capacity()) {
072            int newCapacity = vBuf.position() + 600 * VERTEX_STRIDE;
073
074            Log.put(1, "increasing sprite renderer vertex buffer capacity to " + newCapacity);
075
076            FloatBuffer nBuf = BufferUtils.createFloatBuffer(newCapacity);
077
078            vBuf.flip();
079            nBuf.put(vBuf);
080            vBuf = nBuf;
081        }
082
083        if(flip) {
084            float t = sy1;
085
086            sy1 = sy2;
087            sy2 = t;
088        }
089
090        vBuf.put(dx1);
091        vBuf.put(dy1);
092        vBuf.put(sx1);
093        vBuf.put(sy1);
094        vBuf.put(r);
095        vBuf.put(g);
096        vBuf.put(b);
097        vBuf.put(a);
098
099        vBuf.put(dx1);
100        vBuf.put(dy2);
101        vBuf.put(sx1);
102        vBuf.put(sy2);
103        vBuf.put(r);
104        vBuf.put(g);
105        vBuf.put(b);
106        vBuf.put(a);
107
108        vBuf.put(dx2);
109        vBuf.put(dy2);
110        vBuf.put(sx2);
111        vBuf.put(sy2);
112        vBuf.put(r);
113        vBuf.put(g);
114        vBuf.put(b);
115        vBuf.put(a);
116
117        vBuf.put(dx2);
118        vBuf.put(dy2);
119        vBuf.put(sx2);
120        vBuf.put(sy2);
121        vBuf.put(r);
122        vBuf.put(g);
123        vBuf.put(b);
124        vBuf.put(a);
125
126        vBuf.put(dx2);
127        vBuf.put(dy1);
128        vBuf.put(sx2);
129        vBuf.put(sy1);
130        vBuf.put(r);
131        vBuf.put(g);
132        vBuf.put(b);
133        vBuf.put(a);
134
135        vBuf.put(dx1);
136        vBuf.put(dy1);
137        vBuf.put(sx1);
138        vBuf.put(sy1);
139        vBuf.put(r);
140        vBuf.put(g);
141        vBuf.put(b);
142        vBuf.put(a);
143    }
144
145    public void push(String text, int cw, int ch, int cols, int lineSpacing, int x, int y, float r, float g, float b, float a) {
146        int sx = x;
147
148        for(int i = 0; i != text.length(); i++) {
149            char c = text.charAt(i);
150
151            if(c == '\n') {
152                x = sx;
153                y += lineSpacing + ch;
154            } else {
155                int j = (int)c - (int)' ';
156
157                if(j >= 0 && j < 100) {
158                    int col = j % cols;
159                    int row = j / cols;
160
161                    push(col * cw, row * ch, cw, ch, x, y, cw, ch, r, g, b, a, false);
162                    x += cw;
163                }
164            }
165        }
166    }
167
168    public void endSprite() {
169        int count = vBuf.position() / VERTEX_STRIDE;
170
171        if(count > 0) {
172            vBuf.flip();
173
174            GL15.glBufferData(GL15.GL_ARRAY_BUFFER, vBuf, GL15.GL_STREAM_DRAW);
175            GL11.glDrawArrays(GL11.GL_TRIANGLES, 0, count);
176        }
177    }
178
179    void end() {
180        GL15.glBindBuffer(GL15.GL_ARRAY_BUFFER, 0);
181        shader.end();
182    }
183
184    @Override
185    void destroy() throws Exception {
186        shader.destroy();
187        GL15.glDeleteBuffers(vbo);
188        super.destroy();
189    }
190}
