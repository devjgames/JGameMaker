001package org.game;
002
003import java.util.Vector;
004
005public final class Collider {
006    
007    public float radius = 16;
008    public int groundSlope = 60;
009    public int roofSlope = 50;
010    public final Vec3 velocity = new Vec3();
011
012    private int tested = 0;
013    private final Vec3 f = new Vec3();
014    private final Vec3 u = new Vec3();
015    private final Vec3 r = new Vec3();
016    private final Vec3 o = new Vec3();
017    private final Vec3 d = new Vec3();
018    private final Vec3 p = new Vec3();
019    private final Vec3 c = new Vec3();
020    private final AABB bounds = new AABB();
021    private final Triangle triangle = new Triangle();
022    private boolean onGround = false;
023    private boolean hitRoof = false;
024    private final Mat4 groundMatrix = new Mat4();
025    private final Vec3 delta = new Vec3();
026    private final Vec3 hNormal = new Vec3();
027    private final Vec3 groundNormal = new Vec3();
028    private final Triangle hTriangle = new Triangle();
029    private SceneNode hNode = null;
030    private final Vec3 rPosition = new Vec3();
031    private final float[] time = new float[1];
032    private final Vector<SceneNode> hitNodes = new Vector<>();
033
034    public int getHitNodeCount() {
035        return hitNodes.size();
036    }
037
038    public SceneNode getHitNode(int i) {
039        return hitNodes.get(i);
040    }
041
042    public boolean isOnGround() {
043        return onGround;
044    }
045
046    public boolean didHitRoof() {
047        return hitRoof;
048    }
049
050    public int getTested() {
051        return tested;
052    }
053
054    public boolean intersect(Scene scene, SceneNode root, Vec3 origin, Vec3 direction, float buffer, int mask, float[] time, boolean ignoreBackfaces, Triangle hit) throws Exception {
055        hNode = null;
056        root.traverse((n) -> {
057            bounds.clear();
058            bounds.add(origin);
059            bounds.add(direction.scale(time[0], f).add(origin));
060            if(n.bounds.touches(bounds)) {
061                if(n.collidable) {
062                    OctTree octTree = n.getOctTree(scene);
063
064                    if(octTree != null) {
065                        octTree.traverse((t) -> {
066                            if(t.getBounds().touches(bounds)) {
067                                for(int i = 0; i != t.getTriangleCount(); i++) {
068                                    t.getTriangle(i, triangle);
069                                    if((triangle.tag & mask) != 0) {
070                                        boolean skip = ignoreBackfaces;
071                                        
072                                        if(skip) {
073                                            skip = triangle.n.dot(direction) > 0;
074                                        }
075                                        if(!skip) {
076                                            if(triangle.intersects(origin, direction, buffer, time)) {;
077                                                hit.set(triangle);
078                                                hNode = n;
079                                            }
080                                        }
081                                    }
082                                }
083                                return true;
084                            }
085                            return false;
086                        });
087                    } else {
088                        for(int i = 0; i != n.getTriangleCount(); i++) {
089                            n.getTriangle(scene, i, triangle);
090                            if((triangle.tag & mask) != 0) {
091                                boolean skip = ignoreBackfaces;
092                                        
093                                if(skip) {
094                                    skip = triangle.n.dot(direction) > 0;
095                                }
096                                if(!skip) {
097                                    if(triangle.intersects(origin, direction, buffer, time)) {
098                                        hit.set(triangle);
099                                        hNode = n;
100                                    }
101                                }
102                            }
103                        }
104                    }
105                }
106                return true;
107            }
108            return false;
109        });
110        hitNodes.clear();
111        if(hNode != null) {
112            hitNodes.add(hNode);
113        }
114        return hNode != null;
115    }
116
117    public boolean resolve(Scene scene, SceneNode root, Vec3 position) throws Exception {
118        boolean collided = false;
119
120        hitNodes.clear();
121
122        velocity.scale(Game.getInstance().elapsedTime(), delta);
123        if(delta.length() > radius * 0.5f) {
124            delta.normalize(radius * 0.5f);
125        }
126        groundMatrix.transformNormal(delta);
127        position.add(delta);
128        groundMatrix.toIdentity();
129        onGround = false;
130        groundNormal.toZero();
131        tested = 0;
132        for(int i = 0; i != 3; i++) {
133            hNode = null;
134            hNormal.toZero();
135            bounds.min.set(position).sub(radius, radius, radius);
136            bounds.max.set(position).add(radius, radius, radius);
137            time[0] = radius;
138            root.traverse((n) -> {
139                if(n.bounds.touches(bounds)) {
140                    if(n.collidable) {
141                        OctTree tree = n.getOctTree(scene);
142
143                        if(tree != null) {
144                            tree.traverse((t) -> {
145                                if(t.getBounds().touches(bounds)) {
146                                    for(int j = 0; j != t.getTriangleCount(); j++) {
147                                        if(resolve(t.getTriangle(j, triangle), position, hNormal, rPosition)) {
148                                            hNode = n;
149                                        }
150                                    }
151                                    return true;
152                                }
153                                return false;
154                            });
155                        } else {
156                            for(int j = 0; j != n.getTriangleCount(); j++) {
157                                if(resolve(n.getTriangle(scene, j, triangle), position, hNormal, rPosition)) {
158                                    hNode = n;
159                                }
160                            }
161                        }
162                    }
163                    return true;
164                }
165                return false;
166            });
167            if(hNode != null) {
168                if(Math.acos(Math.max(-0.999f, Math.min(0.999f, hNormal.dot(0, 1, 0)))) < Math.toRadians(groundSlope)) {
169                    groundNormal.add(hNormal);
170                    onGround = true;
171                    velocity.y = 0;
172                }
173                if(Math.acos(Math.max(-0.999f, Math.min(0.999f, hNormal.dot(0, -1, 0)))) < Math.toRadians(roofSlope)) {
174                    hitRoof = true;
175                    velocity.y = 0;
176                }
177                position.set(rPosition);
178                collided = true;
179                hitNodes.add(hNode);
180            } else {
181                break;
182            }
183        }
184        if(onGround) {
185            groundNormal.normalize(u);
186            r.set(1, 0, 0);
187            r.cross(u, f).normalize();
188            u.cross(f, r).normalize();
189            groundMatrix.set(
190                r.x, u.x, f.x, 0,
191                r.y, u.y, f.y, 0,
192                r.z, u.z, f.z, 0,
193                0, 0, 0, 1
194            );
195        }
196        return collided;
197    }
198
199    private boolean resolve(Triangle triangle, Vec3 position, Vec3 hNormal, Vec3 rPosition) {
200        float t = time[0];
201
202        triangle.n.negate(d);
203        o.set(position);
204        if(triangle.intersectsPlane(o, d, time)) {
205            p.set(d).scale(time[0]);
206            o.add(p, p);
207            if(triangle.contains(p, 0)) {
208                hNormal.set(triangle.n);
209                hTriangle.set(triangle);
210                p.add(rPosition.set(hNormal).scale(radius), rPosition);
211                return true;
212            } else {
213                time[0] = t;
214                triangle.closestPoint(o, c);
215                o.sub(c, d);
216                if(d.length() > 0.0000001 && d.length() < time[0]) {
217                    time[0] = d.length();
218                    d.normalize(hNormal);
219                    hTriangle.set(triangle);
220                    c.add(rPosition.set(hNormal).scale(radius), rPosition);
221                    return true;
222                }
223            }
224        }
225        tested++;
226
227        return false;
228    }
