018    private final Vector<SceneNode> renderables = new Vector<>();
019    private final Vector<SceneNode> lights = new Vector<>();

021    private int trianglesRendered = 0;
022    private SceneNode axis = new SceneNode();
023    private final Mat4 matrix = new Mat4();

025    public int getTrianglesRendered() {
026        return trianglesRendered;

029    void render(Scene scene) throws Exception {
030
031        scene.calcBoundsAndTransform();
032        scene.root.traverse((n) -> {
033            n.init();
034            return true;
035        });
036        scene.calcBoundsAndTransform();
037        scene.root.traverse((n) -> {
038            n.start();
039            return true;
040        });
041        scene.calcBoundsAndTransform();
042        scene.root.traverse((n) -> {
043            n.update(scene);
044            return true;
045        });
046        scene.calcBoundsAndTransform();
047
048        if(scene.brush != null) {
049            scene.brush.calcBoundsAndTransform();
050        }
051
052        addNodes(scene);
053
054        renderNodes(scene);
055
056        scene.root.traverse((n) -> {
057            for(int i = 0; i != n.getComponentCount(); i++) {
058                String s = n.getComponent(i).loadSceneName();
059
060                if(s != null) {
061                    scene.setLoadFile(IO.file(AssetManager.getRoot(), s + ".scn"));
062                }
063            }
064            return true;
065        });
066
067        clearNodes();
068    }
069
070    private void addNodes(Scene scene) throws Exception {
071        clearNodes();
072
073        scene.root.traverse((n) -> {
074            if(n.visible) {
075                if(n.emitsLight) {
076                    lights.add(n);
077                }
078                return true;
079            }
080            return false;
081        });
082
083        if(scene.brush != null) {
084            scene.brush.traverse((n) -> {
085                if(n.visible) {
086                    if(n.emitsLight) {
087                        lights.add(n);
088                    }
089                    return true;
090                }
091                return false;
092            });
093        }
094
095        lights.sort((a, b) -> {
096            float da = a.absolutePosition.distance(scene.target);
097            float db = b.absolutePosition.distance(scene.target);
098
099            return Float.compare(da, db);
100        });
101
102        trianglesRendered = 0;
103
104        scene.root.traverse((n) -> {
105            if(n.visible) {
106                if(n.renderable != null || n.getComponentCount() != 0) {
107                    renderables.add(n);
108                } else if(n.emitsLight || n.isLocation || n.drawLines) {
109                    if(scene.isInDesign()) {
110                        renderables.add(n);
111                    }
112                }
113                return true;
114            }
115            return false;
116        });
117
118        if(scene.isInDesign()) {
119            renderables.add(axis);
120        }
121
122        if(scene.brush != null) {
123            scene.brush.traverse((n) -> {
124                if(n.visible) {
125                    if(n.renderable != null || n.getComponentCount() != 0) {
126                        renderables.add(n);
127                    } else if(n.emitsLight || n.isLocation || n.drawLines) {
128                        if(scene.isInDesign()) {
129                            renderables.add(n);
130                        }
131                    }
132                    return true;
133                }
134                return false;
135            });
136        }
137
138        renderables.sort((a, b) -> {
139            if(a.zOrder == b.zOrder) {
140                float da = a.absolutePosition.distance(scene.eye);
141                float db = b.absolutePosition.distance(scene.eye);
142
143                return Float.compare(db, da);
144            } else {
145                return Integer.compare(a.zOrder, b.zOrder);
146            }
147        });
148    }
149
150    private void clearNodes() {
151        renderables.clear();
152        lights.clear();
153        meshes.clear();
154    }
155
156    private void renderNodes(Scene scene) throws Exception {
157        GFX.clear(scene.backgroundColor.x, scene.backgroundColor.y, scene.backgroundColor.z, scene.backgroundColor.w);
158
159        DepthState depthState = null;
160        CullState cullState = null;
161        BlendState blendState = null;
162
163        LineRenderer lineRenderer = Game.getInstance().getRenderer(LineRenderer.class);
164
165        for(SceneNode node : renderables) {
166            if(depthState != node.depthState) {
167                GFX.setDepthState(depthState = node.depthState);
168            }
169            if(cullState != node.cullState) {
170                GFX.setCullState(cullState = node.cullState);
171            }
172            if(blendState != node.blendState) {
173                GFX.setBlendState(blendState = node.blendState);
174            }
175            if(node.renderable != null) {
176                trianglesRendered += node.renderable.render(scene, node, lights);
177            } else if((node.emitsLight || node.isLocation || node.drawLines) && scene.isInDesign()) {
178                lineRenderer.begin(scene.projection, scene.view, matrix.toIdentity());
179                if(node.drawTreeLines) {
180                    drawTreeLines(node, lineRenderer);
181                } else if(node.drawLines) {
182                    for(int i = 0; i < node.getChildCount() - 1; i++) {
183                        Vec3 p1 = node.getChild(i).absolutePosition;
184                        Vec3 p2 = node.getChild(i + 1).absolutePosition;
185
186                        lineRenderer.push(p1.x, p1.y, p1.z, 1, 0.5f, 0, 1, p2.x, p2.y, p2.z, 1, 0.5f, 0, 1);
187                    }
188                }
189                Vec3 p = node.absolutePosition;
190                Vec3 r = node.r;
191                Vec3 u = node.u;
192                Vec3 f = node.f;
193
194                lineRenderer.push(p.x, p.y, p.z, 1, 0, 0, 1, p.x + r.x * 16, p.y + r.y * 16, p.z + r.z * 16, 1, 0, 0, 1);
195                lineRenderer.push(p.x, p.y, p.z, 0, 1, 0, 1, p.x + u.x * 16, p.y + u.y * 16, p.z + u.z * 16, 0, 1, 0, 1);
196                lineRenderer.push(p.x, p.y, p.z, 0, 0, 1, 1, p.x + f.x * 16, p.y + f.y * 16, p.z + f.z * 16, 0, 0, 1, 1);
197                lineRenderer.end();
198            } else if(node == axis) {
199                float l = scene.calcOffset().length() / 6;
200                
201                lineRenderer.begin(scene.projection, scene.view, matrix.toIdentity().translate(scene.target));
202                lineRenderer.push(0, 0, 0, 1, 0, 0, 1, l, 0, 0, 1, 0, 0, 1);
203                lineRenderer.push(0, 0, 0, 0, 1, 0, 1, 0, l, 0, 0, 1, 0, 1);
204                lineRenderer.push(0, 0, 0, 0, 0, 1, 1, 0, 0, l, 0, 0, 1, 1);
205                lineRenderer.end();
206            }
207        }
208
209        SpriteRenderer spriteRenderer = Game.getInstance().getRenderer(SpriteRenderer.class);
210
211        spriteRenderer.begin();
212        scene.root.traverse((n) -> {
213            if(n.visible) {
214                n.renderSprites(spriteRenderer);
215                return true;
216            }
217            return false;
218        });
219        spriteRenderer.end();
220    }
221
222    private void drawTreeLines(SceneNode node, LineRenderer renderer) {
223        Vec3 p1 = node.absolutePosition;
224
225        for(SceneNode child : node) {
226            Vec3 p2 = child.absolutePosition;
227        
228            renderer.push(p1.x, p1.y, p1.z, 1, 0.5f, 0, 1, p2.x, p2.y, p2.z, 1, 0.5f, 0, 1);
229            drawTreeLines(child, renderer);
230        }
231    }

233    public void takeShot(Scene scene) throws Exception {
234        LocalDateTime now = LocalDateTime.now();
235        DateTimeFormatter formatter = DateTimeFormatter.ofPattern("yyyy_MM_dd_HH_mm_ss");
236
237        takeShot(IO.file(IO.file(AssetManager.getRoot(), "shots"), now.format(formatter) + ".png"), scene);
238    }

240    public void takeShot(File file, Scene scene) throws Exception {
241        RenderTarget target = null;
242
243        try {
244            try {
245                addNodes(scene);
246
247                target = new RenderTarget(Game.getInstance().w(), Game.getInstance().h(), ColorFormat.COLOR);
248                target.begin();
249                renderNodes(scene);
250                target.end();
251            } finally {
252                clearNodes();
253            }
254
255            file.getParentFile().mkdirs();
256
257            ByteBuffer buf = ByteBuffer.allocateDirect(target.texture.w * target.texture.h * 4).order(ByteOrder.nativeOrder());
258
259            GL11.glBindTexture(GL11.GL_TEXTURE_2D, target.texture.id);
260            GL11.glGetTexImage(GL11.GL_TEXTURE_2D, 0, GL11.GL_RGBA, GL11.GL_UNSIGNED_BYTE, buf);
261            GL11.glBindTexture(GL11.GL_TEXTURE_2D, 0);
262
263            int[] pixels = new int[target.texture.w * target.texture.h];
264            byte[] rgba = new byte[target.texture.w * target.texture.h * 4];
265
266            buf.get(rgba);
267
268            for(int x = 0; x != target.texture.w; x++) {
269                for(int y = 0; y != target.texture.h; y++) {
270                    int i = (target.texture.h - y - 1) * target.texture.w + x;
271                    int j = y * target.texture.w * 4 + x * 4;
272                    int r = rgba[j++];
273                    int g = rgba[j++];
274                    int b = rgba[j];
275
276                    pixels[i] = 0xFF000000 | ((r << 16) & 0xFF0000) | ((g << 8) & 0xFF00) | (b & 0xFF);
277                }
278            }
279
280            BufferedImage image = new BufferedImage(target.texture.w, target.texture.h, BufferedImage.TYPE_INT_ARGB);
281
282            image.setRGB(0, 0, target.texture.w, target.texture.h, pixels, 0, target.texture.w);
283
284            ImageIO.write(image, "PNG", file);
285        } catch(Exception ex) {
286            Log.put(0, ex);
287        } finally {
288            if(target != null) {
289                target.destroy();
290            }
291        }
292    }
