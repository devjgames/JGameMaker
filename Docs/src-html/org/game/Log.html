001package org.game;
002
003import java.io.ByteArrayOutputStream;
004import java.io.PrintStream;
005
006import javax.swing.JTextArea;
007
008public class Log {
009    
010    public static int level = 2;
011    
012    static boolean paused = false;
013
014    public static void put(int level, Object value) {
015        if(level > Log.level || paused) {
016            return;
017        }
018
019        UIGameEditor editor = UIGameEditor.getInstance();
020
021        if(editor != null) {
022            JTextArea area = editor.getConsoleArea();
023
024            if(value == null) { 
025                area.append("null\n");
026            } else if(value instanceof Exception) {
027                try {
028                    ByteArrayOutputStream out = null;
029        
030                    try {
031                        PrintStream stream = null;
032
033                        out = new ByteArrayOutputStream(1000);
034                        try {
035                            ((Exception)value).printStackTrace(stream = new PrintStream(out));
036
037                            area.append(new String(out.toByteArray()) + "\n");
038                        } finally {
039                            if(stream != null) {
040                                stream.close();
041                            }
042                        }
043                    } finally {
044                        if(out != null) {
045                            out.close();
046                        }
047                    }
048                } catch(Exception ex) {
049                    ex.printStackTrace();
050                }
051            } else {
052                area.append(value + "\n");
053            }
054            area.setCaretPosition(area.getDocument().getLength());
055        } 
056        if(value == null) {
057            System.out.println("null");
058        } else if(value instanceof Exception) {
059            ((Exception)value).printStackTrace();
060        } else {
061            System.out.println(value);
062        }
063    }
064}
