001package org.game;
002
003import java.io.File;
004import java.util.Vector;
005
006public final class ParticleSystem implements Renderable {
007
008    public Texture texture = null;
009    public final Vec3 emitPosition = new Vec3();
010
011    private float[] particles;
012    private float[] temp;
013    private int count;
014    private final Mat4 m = new Mat4();
015    private final Vec3 r = new Vec3();
016    private final Vec3 u = new Vec3();
017    private final Vec3 f = new Vec3();
018    private float time = 0;
019    private int stackTop = 0;
020    private final int maxParticles;
021    private final Vector<Vertex> vertices = new Vector<>();
022    private final Vector<Integer> indices = new Vector<>();
023    private final AABB bounds = new AABB();
024    private int particleCount = 0;
025    private File file;
026
027    public ParticleSystem(int maxParticles) throws Exception {
028        for(int i = 0; i != maxParticles * 4; i++) {
029            vertices.add(new Vertex());
030        }
031        for(int i = 0; i != maxParticles * 4; i += 4) {
032            indices.add(i);
033            indices.add(i + 1);
034            indices.add(i + 2);
035            indices.add(i + 2);
036            indices.add(i + 3);
037            indices.add(i);
038        }
039        this.maxParticles = maxParticles;
040
041        count = 0;
042        particles = new float[maxParticles * 20];
043        temp = new float[maxParticles * 20];
044    }
045
046    ParticleSystem(ParticleSystem particles) throws Exception {
047        if(texture != null) {
048            try {
049                texture = Game.getInstance().getAssets().load(texture.file);
050            } catch(Exception ex) {
051                Log.put(0, ex);
052            }
053        }
054
055        maxParticles = particles.maxParticles;
056        for(int i = 0; i != maxParticles * 4; i++) {
057            vertices.add(new Vertex());
058        }
059        this.indices.addAll(particles.indices);
060
061        count = 0;
062        this.particles = new float[maxParticles * 20];
063        temp = new float[maxParticles * 20];
064
065        Utils.copy(particles, this);
066
067        file = particles.file;
068    }
069
070    @Override
071    public File getFile() {
072        return file;
073    }
074
075    @Override
076    public int getTriangleCount() {
077        return particleCount * 2;
078    }
079
080    @Override
081    public Triangle getTriangle(Scene scene, SceneNode node, int i, Triangle triangle) {
082        i *= 3;
083        triangle.p1.set(vertices.get(indices.get(i + 0)).position);
084        triangle.p2.set(vertices.get(indices.get(i + 1)).position);
085        triangle.p3.set(vertices.get(indices.get(i + 2)).position);
086        return triangle.calcPlane();
087    }
088
089    public AABB getBounds() {
090        return bounds;
091    }
092
093    @Override
094    public void update(Scene scene, SceneNode node) throws Exception {
095        time = Game.getInstance().totalTime();
096        int count = 0;
097        for(int i = 0; i != this.count; i += 20) {
098            float n = (time - particles[i + 18]) / particles[i + 19];
099            if(n <= 1) {
100                for(int j = 0; j != 20; j++) {
101                    temp[count++] = particles[i + j];
102                }
103            }
104        }
105        float[] t = particles;
106        particles = temp;
107        temp = t;
108        this.count = count;
109
110        m.set(scene.view).mul(node.model).transpose();
111        r.set(m.m00, m.m10, m.m20).normalize();
112        u.set(m.m01, m.m11, m.m21).normalize();
113        f.set(m.m02, m.m12, m.m22).normalize();
114
115        bounds.clear();
116        stackTop = 0;
117        particleCount = 0;
118        for(int i = 0; i != count; particleCount++) {
119            float vX = particles[i++];
120            float vY = particles[i++];
121            float vZ = particles[i++];
122            float pX = particles[i++];
123            float pY = particles[i++];
124            float pZ = particles[i++];
125            float sR = particles[i++];
126            float sG = particles[i++];
127            float sB = particles[i++];
128            float sA = particles[i++];
129            float eR = particles[i++];
130            float eG = particles[i++];
131            float eB = particles[i++];
132            float eA = particles[i++];
133            float sX = particles[i++];
134            float sY = particles[i++];
135            float eX = particles[i++];
136            float eY = particles[i++];
137            float s = time - particles[i++];
138            float n = s / particles[i++];
139            float cR = sR + n * (eR - sR);
140            float cG = sG + n * (eG - sG);
141            float cB = sB + n * (eB - sB);
142            float cA = sA + n * (eA - sA);
143            float x = (sX + n * (eX - sX)) * 0.5f;
144            float y = (sY + n * (eY - sY)) * 0.4f;
145
146            pX += s * vX;
147            pY += s * vY;
148            pZ += s * vZ;
149            push(pX - r.x * x - u.x * y, pY - r.y * x - u.y * y, pZ - r.z * x - u.z * y, 0, 0, f.x, f.y, f.z, cR, cG, cB, cA);
150            push(pX + r.x * x - u.x * y, pY + r.y * x - u.y * y, pZ + r.z * x - u.z * y, 1, 0, f.x, f.y, f.z, cR, cG, cB, cA);
151            push(pX + r.x * x + u.x * y, pY + r.y * x + u.y * y, pZ + r.z * x + u.z * y, 1, 1, f.x, f.y, f.z, cR, cG, cB, cA);
152            push(pX - r.x * x + u.x * y, pY - r.y * x + u.y * y, pZ - r.z * x + u.z * y, 0, 1, f.x, f.y, f.z, cR, cG, cB, cA);
153            bounds.add(pX - r.x * x - u.x * y, pY - r.y * x - u.y * y, pZ - r.z * x - u.z * y);
154            bounds.add(pX + r.x * x - u.x * y, pY + r.y * x - u.y * y, pZ + r.z * x - u.z * y);
155            bounds.add(pX + r.x * x + u.x * y, pY + r.y * x + u.y * y, pZ + r.z * x + u.z * y);
156            bounds.add(pX - r.x * x + u.x * y, pY - r.y * x + u.y * y, pZ - r.z * x + u.z * y);
157        }
158    }
159
160    private void push(float x, float y, float z, float s, float t, float nx, float ny, float nz, float r, float g, float b, float a) {
161        Vertex v = vertices.get(stackTop++);
162
163        v.position.set(x, y, z);
164        v.textureCoordinate.set(s, t);
165        v.normal.set(nx, ny, nz);
166        v.color.set(r, g, b, a);
167    }
168    
169    public void emit(Particle particle) {
170        if(count != particles.length) {
171            particles[count++] = particle.velocityX;
172            particles[count++] = particle.velocityY;
173            particles[count++] = particle.velocityZ;
174            particles[count++] = particle.positionX + emitPosition.x;
175            particles[count++] = particle.positionY + emitPosition.y;
176            particles[count++] = particle.positionZ + emitPosition.z;
177            particles[count++] = particle.startR;
178            particles[count++] = particle.startG;
179            particles[count++] = particle.startB;
180            particles[count++] = particle.startA;
181            particles[count++] = particle.endR;
182            particles[count++] = particle.endG;
183            particles[count++] = particle.endB;
184            particles[count++] = particle.endA;
185            particles[count++] = particle.startX;
186            particles[count++] = particle.startY;
187            particles[count++] = particle.endX;
188            particles[count++]  = particle.endY;
189            particles[count++] = Game.getInstance().totalTime();
190            particles[count++] = particle.lifeSpan;
191        }
192    }
193
194    @Override
195    public int render(Scene scene, SceneNode node, Vector<SceneNode> lights) throws Exception {
196        LightRenderer renderer = Game.getInstance().getRenderer(LightRenderer.class);
197
198        renderer.begin(scene, node, lights, texture, null);
199        renderer.push(vertices, indices, particleCount * 6);
200        renderer.end();
201
202        return particleCount * 2;
203    }
204
205    @Override
206    public Renderable newInstance() throws Exception {
207        return new ParticleSystem(this);
208    }
209}
