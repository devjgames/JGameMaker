035    private boolean running = true;
036    private ResourceManager resources = new ResourceManager();
037    private AssetManager assets;
038    private HashSet<Renderer> renderers = new HashSet<>();
039    private double lastTime = 0;
040    private double totalTime = 0;
041    private double elapsedTime = 0;

043    private int frames = 0;
044    private int fps = 0;

051    Game(int w, int h, Canvas canvas, GameLoop loop) throws Exception {
052        instance = this;
053
054        String os = System.getProperty("os.name").toLowerCase();
055
056        if(!os.startsWith("windows")) {
057            reverseY = true;
058        }
059
060        assets = resources.manage(new AssetManager());
061
062        this.loop = loop;
063
064        if(canvas != null) {
065            this.canvas = canvas;
066
067            Thread thread = new Thread(new Runnable() {
068
069                @Override
070                public void run() {
071                    try {
072                        gameLoop(null, false);
073                    } catch(Exception ex) {
074                        Log.put(0, ex);
075                    }
076                }
077
078            });
079            thread.setPriority(Thread.MIN_PRIORITY);
080            thread.start();
081        } else {
082            DisplayMode[] modes = Display.getAvailableDisplayModes();
083            DisplayMode mode = new DisplayMode(w, h);
084            boolean fs = false;
085            Vector<Object> vModes = new Vector<>();
086
087            for(DisplayMode iMode : modes) {
088                if(iMode.isFullscreenCapable() && iMode.getBitsPerPixel() == 32) {
089                    vModes.add(iMode);
090                }
091            }
092
093            if(!vModes.isEmpty()) {
094                Object[] aryModes = new Object[vModes.size()];
095
096                for(int i = 0; i != aryModes.length; i++) {
097                    aryModes[i] = vModes.get(i);
098                }
099
100                Object r = JOptionPane.showInputDialog(null, "Mode", "Mode", JOptionPane.QUESTION_MESSAGE, null, aryModes, aryModes[0]);
101
102                if(r != null) {
103                    fs = true;
104                    mode = (DisplayMode)r;
105                }
106            }
107
108            gameLoop(mode, fs);
109        }
110    }
111
112    private void gameLoop(DisplayMode mode, boolean fs) throws Exception {
113        try {
114            if(mode != null) {
115                Display.setDisplayMode(mode);
116                Display.create();
117            } else if(canvas != null) {
118                Display.setParent(canvas);
119                Display.create(
120                    new PixelFormat()
121                        .withBitsPerPixel(32)
122                        .withAlphaBits(8)
123                );
124            } else {
125                throw new Exception("invalid game loop parameters");
126            }
127            Display.makeCurrent();
128            if(fs) {
129                Display.setFullscreen(true);
130            }
131            Mouse.create();
132            Keyboard.create();
133
134            Log.put(0, "GL Version = '" + getGLVersion() + "'");
135
136            loop.init();
137
138            while(running) {
139                Display.makeCurrent();
140
141                GL11.glViewport(0, 0, w(), h());
142                loop.render();
143                GFX.checkError("display");
144                Display.update(true);
145                Display.sync(60);
146                tick();
147
148                if(keyDown(Keys.KEY_ESCAPE) && canvas == null) {
149                    break;
150                }
151                if(Display.isCloseRequested()) {
152                    break;
153                }
154            }
155        } finally {
156            destroy();
157        }
158    }

168    @SuppressWarnings("unchecked")
169    <T extends Renderer> T getRenderer(Class<? extends Renderer> cls) throws Exception {
170        for(Renderer renderer : renderers) {
171            if(cls.isAssignableFrom(renderer.getClass())) {
172                return (T)renderer;
173            }
174        }
175
176        Log.put(0, "creating renderer - " + cls.getName() + " ...");
177
178        Renderer renderer = null;
179
180        try {
181            renderer = resources.manage((T)cls.getConstructors()[0].newInstance());
182        } catch(InvocationTargetException ex) {
183            Log.put(0, ex);
184            throw ex;
185        }
186
187        renderers.add(renderer);
188
189        return (T)renderer;
190    }

198    }

200    public int mouseY() {
201        int y = Mouse.getY();
202
203        if(reverseY) {
204            y = h() - y - 1;
205        }
206        return y;
207    }

209    public void setMouseGrabbed(boolean grabbed) {
210        if(grabbed) {
211            Mouse.setGrabbed(true);
212            Mouse.poll();
213            centerMouse();
214        } else {
215            Mouse.setGrabbed(false);
216        }
217    }
218
219    public int dX() {
220        return dX;
221    }
222
223    public int dY() {
224        int y = dY;
225
226        if(reverseY) {
227            y = -y;
228        }
229        return y;
230    }

264    public void resetTimer() {
265        lastTime = System.nanoTime() / 1000000000.0;
266        totalTime = 0;
267        elapsedTime = 0;
268        seconds = 0;
269        frames = 0;
270        fps = 0;
271    }
272
273    void addWindowListener(JFrame frame) {
274        frame.setDefaultCloseOperation(JFrame.DO_NOTHING_ON_CLOSE);
275
276        frame.addWindowListener(new WindowAdapter() {
277
278            @Override
279            public void windowClosing(WindowEvent e) {
280                running = false;
281                frame.dispose();
282            }
283            
284        });
285    }
286
287    private void destroy() {
288        try {
289            Log.put(0, Resource.getInstances() + " instance(s)");
290            if(Display.isCreated()) {
291                resources.destroy();
292                Display.destroy();
293            }
294            if(Mouse.isCreated()) {
295                Mouse.destroy();
296            }
297            if(Keyboard.isCreated()) {
298                Keyboard.destroy();
299            }
300            Log.put(0, Resource.getInstances() + " instance(s)");
301        } catch(Exception ex) {
302            Log.put(0, ex);
303        }
304    }
305
306    private void tick() {
307        double nowTime = System.nanoTime() / 1000000000.0;
308        elapsedTime = nowTime - lastTime;
309        lastTime = nowTime;
310        seconds += elapsedTime;
311        totalTime += elapsedTime;
312        frames++;
313        if (seconds >= 1) {
314            fps = frames;
315            frames = 0;
316            seconds = 0;
317        }
318        dX = Mouse.getDX();
319        dY = Mouse.getDY();
320        if(Mouse.isGrabbed()) {
321            centerMouse();
322        }
323    }
324
325    private void centerMouse() {
326        Mouse.setCursorPosition(Display.getX() + Display.getWidth() / 2, Display.getY() + Display.getHeight() / 2);
327    }
