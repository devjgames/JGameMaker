001package org.game;
002
003import org.lwjgl.opengl.GL11;
004
005public final class GFX {
006    
007    public static void setDepthState(DepthState state) {
008        if(state == DepthState.NONE) {
009            GL11.glDisable(GL11.GL_DEPTH_TEST);
010            GL11.glDepthMask(false);
011        } else {
012            GL11.glEnable(GL11.GL_DEPTH_TEST);
013            if(state == DepthState.READONLY) {
014                GL11.glDepthMask(false);
015            } else {
016                GL11.glDepthMask(true);
017            }
018        }
019    }
020
021    public static void setBlendState(BlendState state) {
022        if(state == BlendState.OPAQUE) {
023            GL11.glDisable(GL11.GL_BLEND);
024        } else {
025            GL11.glEnable(GL11.GL_BLEND);
026            if(state == BlendState.ADDITIVE) {
027                GL11.glBlendFunc(GL11.GL_ONE, GL11.GL_ONE);
028            } else {
029                GL11.glBlendFunc(GL11.GL_SRC_ALPHA, GL11.GL_ONE_MINUS_SRC_ALPHA);
030            }
031        }
032    }
033
034    public static void setCullState(CullState state) {
035        if(state == CullState.NONE) {
036            GL11.glDisable(GL11.GL_CULL_FACE);
037        } else {
038            GL11.glEnable(GL11.GL_CULL_FACE);
039            if(state == CullState.BACK) {
040                GL11.glCullFace(GL11.GL_BACK);
041            } else {
042                GL11.glCullFace(GL11.GL_FRONT);
043            }
044        }
045    }
046
047    public static void clear(float r, float g, float b, float a) {
048        GL11.glEnable(GL11.GL_POLYGON_OFFSET_FILL);
049        GL11.glPolygonOffset(1, 1);
050        setDepthState(DepthState.READWRITE);
051        setCullState(CullState.BACK);
052        setBlendState(BlendState.OPAQUE);
053        GL11.glClearColor(r, g, b, a);
054        GL11.glClear(GL11.GL_COLOR_BUFFER_BIT | GL11.GL_DEPTH_BUFFER_BIT);
055    }
056
057    private static boolean hasError = false;
058
059    public static void checkError(String tag) {
060        if(!hasError) {
061            int code = GL11.glGetError();
062
063            if(code != GL11.GL_NO_ERROR) {
064                hasError = true;
065                Log.put(0, tag + ":" + code);
066            }
067        }
068    }
069
070    private static final Mat4 m = new Mat4();
071    private static final Vec4 v = new Vec4();
072
073    public static Vec3 unproject(float x, float y, float z, float vx, float vy, float vw, float vh, Mat4 projection, Mat4 view, Vec3 p) {
074        m.set(projection).mul(view).invert();
075
076        v.x = 2 * (x - vx) / vw - 1;
077        v.y = 2 * (y - vy) / vh - 1;
078        v.z = 2 * z - 1;
079        v.w = 1;
080        m.transform(v);
081        v.scale(1 / v.w);
082
083        return p.set(v.x, v.y, v.z);
084    }
085}
