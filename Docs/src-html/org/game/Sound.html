001package org.game;
002
003import javax.sound.sampled.*;
004
005import java.io.*;
006
007public final class Sound extends Resource {
008
009    public final File file;
010
011    private Clip clip;
012    private FloatControl gain;
013
014    public Sound(File file) throws Exception {
015        this.file = file;
016        AudioInputStream stream = AudioSystem.getAudioInputStream(file);
017        clip = AudioSystem.getClip();
018        clip.open(stream);
019        gain = (FloatControl)clip.getControl(FloatControl.Type.MASTER_GAIN);
020        stream.close();
021    }
022
023    public void setVolume(float volume) {
024        if(gain != null) {
025            volume = Math.max(0, Math.min(1, volume));
026            gain.setValue(gain.getMinimum() + volume * (gain.getMaximum() - gain.getMinimum()));
027        }
028    }
029
030    public void play(boolean looping) {
031        if(!clip.isActive()) {
032            clip.stop();
033            clip.setFramePosition(0);
034            clip.start();
035            if(looping) {
036                clip.loop(Clip.LOOP_CONTINUOUSLY);
037            } else {
038                clip.loop(0);
039            }
040        }
041    }
042
043    public void stop() {
044        clip.stop();
045    }
046
047    @Override
048    void destroy() throws Exception {
049        clip.stop();
050        clip.close();
051        super.destroy();
052    }
053}
